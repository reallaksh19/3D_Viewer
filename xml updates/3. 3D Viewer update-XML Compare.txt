# Source-level findings

## 1. Elbow rendering

`viewer-3d.js` already has `createTubeFromCurve(...)`, so elbow/bend can be improved without adding a new rendering library. The issue is likely in how bend control point / centre point is used. XML import already tries to set `centrePoint` for BEND using XML bend metadata and near node.

Fix direction:

```text
BEND / ELBOW renderer must prefer:
1. points[0]
2. centrePoint
3. points[1]

and render a real tube arc, not a straight pipe fallback.
```

## 2. Reducer rendering

`viewer-3d.js` has cylinder/disc helpers and REDUCER color, but reducer must be rendered as a tapered frustum, not a normal pipe.

Fix direction:

```text
REDUCER-CONCENTRIC → tapered frustum from bore1 to bore2
REDUCER-ECCENTRIC  → offset tapered frustum if offset data exists
```

## 3. Missing GUI / LIM / hanger / X/Z restraints

Both renderer and XML support builder have partial support classification.

Current gaps:

```text
LIM / LIMIT STOP is not consistently classified as its own support kind.
HANGER may be mixed with guide/spring.
X/Z directional restraints need explicit visual mapping.
```

`xml-support-builder.js` already passes direction and axis metadata, so the renderer can use those values instead of guessing.

```js
function _supportKindFromBlock(blockCode) {
  const b = String(blockCode || '').toUpperCase().trim();
  if (!b) return 'UNK';

  if (b === 'CA100') return 'GDE';
  if (b === 'CA150') return 'RST';
  if (b === 'CA250') return 'RST';
  if (b === 'CA300') return 'ANC';

  // Limit / line stop family.
  if (b === 'CA350') return 'LIM';

  // Optional future support blocks.
  if (b === 'CA400') return 'HANG';
  if (b === 'CA450') return 'X_RESTRAINT';

  return 'UNK';
```

```js
function _supportKindFromDofs(dofs, verticalAxis = 'Y') {
  const set = new Set(_uniqueInts(dofs));
  if (set.size === 0) return 'UNK';

  const verticalDOF = verticalAxis === 'Z' ? 3 : 2;
  const xDOF = 1;
  const zDOF = verticalAxis === 'Z' ? 2 : 3;

  if (set.size >= 5 || (set.has(1) && set.has(2) && set.has(3))) {
    return 'ANC';
  }

  if (set.size === 1 && set.has(verticalDOF)) {
    return 'RST';
  }

  if (set.size === 1 && set.has(xDOF)) {
    return 'X_RESTRAINT';
  }

  if (set.size === 1 && set.has(zDOF)) {
    return 'Z_RESTRAINT';
  }

  if (set.has(xDOF) && set.has(zDOF) && !set.has(verticalDOF)) {
    return 'GDE';
  }

  if (set.has(xDOF) || set.has(zDOF)) {
    return 'GDE';
  }

  if (set.has(verticalDOF)) {
    return 'RST';
  }

  return 'UNK';
}
```

```js
function _semanticDirection(axis, supportKind, opts = {}) {
  const kind = String(supportKind || '').toUpperCase();

  if (kind === 'X_RESTRAINT') return 'EAST';
  if (kind === 'Z_RESTRAINT') return 'NORTH';
  if (kind === 'HANG') return 'DOWN';

  if (!axis) return '';

  const verticalAxisName = String(opts.verticalAxis || 'Y').toUpperCase();
  const up = verticalAxisName === 'Z'
    ? { x: 0, y: 0, z: 1 }
    : { x: 0, y: 1, z: 0 };

  const verticalness = _absDot(axis, up);

  if (kind === 'RST' || kind === 'SPR') {
    return _dot(axis, up) >= 0 ? 'UP' : 'DOWN';
  }

  if (kind === 'LIM' || kind === 'GDE' || kind === 'ANC') {
    if (verticalness < 0.75) return _cardinalFromAxis(axis, opts);
    return _dot(axis, up) >= 0 ? 'UP' : 'DOWN';
  }

  if (verticalness < 0.75) {
    return _cardinalFromAxis(axis, opts);
  }

  return _dot(axis, up) >= 0 ? 'UP' : 'DOWN';
}
```

```js
// ── X1: Professional component renderer helpers ─────────────────────

function _numSafe(value, fallback = 0) {
    const n = Number(value);
    return Number.isFinite(n) ? n : fallback;
}

function _firstFiniteNumber(...values) {
    for (const value of values) {
        const n = Number(value);
        if (Number.isFinite(n)) return n;
    }
    return null;
}

function _componentAttr(comp, ...keys) {
    const attrs = comp?.attributes || {};
    const src = comp?.source || {};

    for (const key of keys) {
        if (attrs[key] !== undefined && attrs[key] !== null && attrs[key] !== '') return attrs[key];
        if (src[key] !== undefined && src[key] !== null && src[key] !== '') return src[key];
    }

    return '';
}

function _componentText(comp) {
    return [
        comp?.type,
        comp?.fixingAction,
        comp?.id,
        _componentAttr(comp, 'SKEY', 'SUPPORT_KIND', 'SUPPORT_NAME', 'SUPPORT_TAG', 'SUPPORT_DIRECTION'),
        _componentAttr(comp, 'SUPPORT_DESC', 'SUPPORT_DOFS', 'AXIS_COSINES', 'PIPE_AXIS_COSINES'),
        _componentAttr(comp, 'COMPONENT-IDENTIFIER', 'COMPONENT-ATTRIBUTE1', 'COMPONENT-ATTRIBUTE2'),
    ].map(v => String(v || '').toUpperCase()).join(' ');
}

function _componentPoint(comp, index = 0) {
    const p = comp?.points?.[index] || null;
    return p ? mapCoord(p) : null;
}

function _componentCoord(comp) {
    const p = comp?.coOrds || comp?.coords || comp?.coord || comp?.points?.[0] || null;
    return p ? mapCoord(p) : null;
}

function _componentCentrePoint(comp) {
    const p = comp?.centrePoint || comp?.centerPoint || comp?.cp || comp?.source?.centrePoint || comp?.source?.centerPoint || null;
    return p ? mapCoord(p) : null;
}

function _componentBoreDiameter(comp, fallback = 100) {
    return Math.max(1, _firstFiniteNumber(
        comp?.bore,
        comp?.od,
        comp?.diameter,
        comp?.points?.[0]?.bore,
        comp?.points?.[1]?.bore,
        _componentAttr(comp, 'BORE', 'OD', 'DIAMETER', 'NOMINAL_DIAMETER', 'NPS_MM')
    ) ?? fallback);
}

function _componentEndBoreDiameter(comp, index, fallback = null) {
    const point = comp?.points?.[index] || {};
    const suffix = index === 0 ? '1' : '2';

    return Math.max(1, _firstFiniteNumber(
        point.bore,
        point.od,
        point.diameter,
        _componentAttr(comp, `BORE${suffix}`, `BORE_${suffix}`, `OD${suffix}`, `OD_${suffix}`),
        _componentAttr(comp, index === 0 ? 'LARGE_BORE' : 'SMALL_BORE'),
        fallback
    ) ?? fallback ?? _componentBoreDiameter(comp));
}

function _radiusFromDiameter(diameter, fallback = 50) {
    return Math.max(1, _numSafe(diameter, fallback) / 2);
}

function _makeProfessionalMaterial(color, materialOverride = null) {
    return materialOverride || new THREE.MeshStandardMaterial({
        color,
        roughness: 0.34,
        metalness: 0.14,
    });
}

function createDiscRing(pos, normal, radius, color, materialOverride = null, thickness = null) {
    const t = thickness ?? Math.max(2, radius * 0.08);
    return createDisc(pos, normal, radius, t, color, materialOverride);
}

function createReducerFrustum(startVec, endVec, radius1, radius2, color, materialOverride = null) {
    const diff = new THREE.Vector3().subVectors(endVec, startVec);
    const length = diff.length();
    if (length < 0.1) return null;

    const mat = _makeProfessionalMaterial(color, materialOverride);
    const geo = new THREE.CylinderGeometry(
        Math.max(1, radius2),
        Math.max(1, radius1),
        length,
        28,
        1,
        false
    );

    const mesh = new THREE.Mesh(geo, mat);
    const mid = new THREE.Vector3().addVectors(startVec, endVec).multiplyScalar(0.5);
    const quat = new THREE.Quaternion().setFromUnitVectors(
        new THREE.Vector3(0, 1, 0),
        diff.clone().normalize()
    );

    mesh.position.copy(mid);
    mesh.quaternion.copy(quat);
    return mesh;
}

function createReducerMeshFromComponent(comp, color, materialOverride = null) {
    const p1 = _componentPoint(comp, 0);
    const p2 = _componentPoint(comp, 1);
    if (!p1 || !p2) return null;

    const dia1 = _componentEndBoreDiameter(comp, 0, _componentBoreDiameter(comp));
    const dia2 = _componentEndBoreDiameter(comp, 1, Math.max(1, dia1 * 0.65));
    const r1 = _radiusFromDiameter(dia1);
    const r2 = _radiusFromDiameter(dia2);

    const group = new THREE.Group();
    group.userData.rendererKind = 'PROFESSIONAL_REDUCER';

    const body = createReducerFrustum(p1, p2, r1, r2, color, materialOverride);
    if (body) group.add(body);

    const axis = new THREE.Vector3().subVectors(p2, p1).normalize();
    const ring1 = createDiscRing(p1, axis, r1 * 1.06, color, materialOverride);
    const ring2 = createDiscRing(p2, axis, r2 * 1.06, color, materialOverride);
    if (ring1) group.add(ring1);
    if (ring2) group.add(ring2);

    return group;
}

function createElbowMeshFromComponent(comp, color, materialOverride = null) {
    const p1 = _componentPoint(comp, 0);
    const p2 = _componentPoint(comp, 1);
    const cp = _componentCentrePoint(comp);
    const radius = _radiusFromDiameter(_componentBoreDiameter(comp));

    if (!p1 || !p2) return null;

    const mat = _makeProfessionalMaterial(color, materialOverride);
    const group = new THREE.Group();
    group.userData.rendererKind = cp ? 'PROFESSIONAL_ELBOW_ARC' : 'PROFESSIONAL_ELBOW_FALLBACK';

    if (cp && cp.distanceTo(p1) > 0.1 && cp.distanceTo(p2) > 0.1) {
        const curve = new THREE.QuadraticBezierCurve3(p1, cp, p2);
        const tube = createTubeFromCurve(curve, radius, color, 36, 16, mat);
        if (tube) group.add(tube);

        const startDir = new THREE.Vector3().subVectors(cp, p1).normalize();
        const endDir = new THREE.Vector3().subVectors(p2, cp).normalize();

        const ring1 = createDiscRing(p1, startDir, radius * 1.06, color, mat);
        const ring2 = createDiscRing(p2, endDir, radius * 1.06, color, mat);
        if (ring1) group.add(ring1);
        if (ring2) group.add(ring2);

        const cpMarker = createSphere(cp, Math.max(2, radius * 0.18), 0xffffff, null);
        cpMarker.userData.nonSelectableHelper = true;
        group.add(cpMarker);

        return group;
    }

    // Fallback: do not disappear; draw a short straight body with end rings.
    const fallback = createCylinder(p1, p2, radius, color, mat);
    if (fallback) group.add(fallback);

    const axis = new THREE.Vector3().subVectors(p2, p1).normalize();
    const ring1 = createDiscRing(p1, axis, radius * 1.06, color, mat);
    const ring2 = createDiscRing(p2, axis, radius * 1.06, color, mat);
    if (ring1) group.add(ring1);
    if (ring2) group.add(ring2);

    return group;
}

function _normalizeSupportKindForRenderer(comp) {
    const text = _componentText(comp);
    const attrKind = String(_componentAttr(comp, 'SUPPORT_KIND', 'SUPPORT-TYPE', 'SUPPORT_TYPE') || '').toUpperCase();

    const combined = `${attrKind} ${text}`;

    if (/\bANC\b|\bANCHOR\b|\bFIX(ED)?\b|\bRIGID\b/.test(combined)) return 'ANCHOR';
    if (/\bHANG\b|\bHANGER\b|\bSPRING\s+HANGER\b|\bROD\s+HANGER\b/.test(combined)) return 'HANGER';
    if (/\bLIM\b|\bLIMIT\b|\bLIMIT\s+STOP\b|\bLINE\s+STOP\b|\bSTOP\b|\bSTP\b/.test(combined)) return 'LIMIT_STOP';
    if (/\bX_RESTRAINT\b|\bX\s*RESTRAINT\b|\bX-REST\b|\bREST-X\b/.test(combined)) return 'X_RESTRAINT';
    if (/\bZ_RESTRAINT\b|\bZ\s*RESTRAINT\b|\bZ-REST\b|\bREST-Z\b/.test(combined)) return 'Z_RESTRAINT';
    if (/\bGDE\b|\bGUI\b|\bGUIDE\b|\bSLIDE\b|\bSLID\b/.test(combined)) return 'GUIDE';
    if (/\bRST\b|\bREST\b|\+Y\b|\bY\s*SUPPORT\b/.test(combined)) return 'REST';
    if (/\bSPR\b|\bSPRING\b/.test(combined)) return 'SPRING';

    return 'UNKNOWN';
}

function _supportAxisFromComponent(comp) {
    const direction = String(_componentAttr(comp, 'SUPPORT_DIRECTION', 'SUPPORT-DIRECTION') || '').toUpperCase();
    const fromDirection = _axisFromSupportDirection(direction);
    if (fromDirection) return fromDirection;

    const fromCosines = _axisFromCosinesText(_componentAttr(comp, 'AXIS_COSINES', 'AXIS-COSINES'));
    if (fromCosines) return fromCosines;

    return null;
}

function _pipeAxisFromComponent(comp) {
    return _axisFromCosinesText(_componentAttr(comp, 'PIPE_AXIS_COSINES', 'PIPE-AXIS-COSINES'));
}

function _defaultPipeAxis(comp) {
    const a = _componentPoint(comp, 0);
    const b = _componentPoint(comp, 1);
    if (a && b && a.distanceTo(b) > 0.1) {
        return new THREE.Vector3().subVectors(b, a).normalize();
    }
    return new THREE.Vector3(1, 0, 0);
}

    const supportAxis = _supportAxisFromComponent(comp);
    const pipeAxis = _pipeAxisFromComponent(comp) || _defaultPipeAxis(comp);

            frame.vertical.clone().multiplyScalar(-scale * 0.55),

        parts.push(block);
    } else if (kind === 'ANCHOR') {
        const anchor = createBox(new THREE.Vector3(0, 0, 0), scale * 0.55, color, true, mat);
        parts.push(anchor);

        parts.push(createArrowBetween(new THREE.Vector3(0, 0, 0), frame.vertical.clone().multiplyScalar(scale * 1.2), color, mat, scale));
        parts.push(createArrowBetween(new THREE.Vector3(0, 0, 0), frame.lateral.clone().multiplyScalar(scale * 1.2), color, mat, scale));
        parts.push(createArrowBetween(new THREE.Vector3(0, 0, 0), frame.pipe.clone().multiplyScalar(scale * 1.2), color, mat, scale));
    } else {
        const fallback = createBox(new THREE.Vector3(0, 0, 0), scale * 0.4, color, true, mat);
        parts.push(fallback);
    }

    const assembly = createSupportAssembly(pos, parts);
    assembly.userData.rendererKind = `PROFESSIONAL_SUPPORT_${kind}`;
    assembly.userData.supportKind = kind;
    return assembly;
}
```

```js
export async function importXmlCompareDataset(file, datasetId) {
  if (!file) {
    throw new Error(`XML ${datasetId}: no file selected.`);
  }

  const name = String(file.name || '').toLowerCase();
  if (!name.endsWith('.xml')) {
    throw new Error(`XML ${datasetId}: expected .xml file, got "${file.name}".`);
  }

  const log = [];
  const result = await importFromRawFile(file, state, log);

  if (!result?.ok || !result.directPcfData) {
    throw new Error(result?.message || `XML ${datasetId}: import failed.`);
  }

    sourceFileName: file.name,

    diagnostics: log,
    directPcfData: result.directPcfData,
  };
}

export function xmlCompareSummary(dataset) {

    byType.set(type, (byType.get(type) || 0) + 1);
  }

  return {
    datasetId: dataset?.id || '',
    fileName: dataset?.sourceFileName || '',
    componentCount: components.length,
    byType: [...byType.entries()].sort((a, b) => a[0].localeCompare(b[0])),
  };
}
```

```js
/**
 * XmlCompareRenderer.js
 *
 * X2 renderer:
 * - Same canvas, same scene.
 * - XML A and XML B as separate groups.
 * - XML B uses a different palette and semi-transparent material.
 */

import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

function num(value, fallback = 0) {
  const n = Number(value);
  return Number.isFinite(n) ? n : fallback;
}

function mapCoord(p) {
  if (!p) return null;

  // Keep same mapping convention as normal PcfViewer3D:
  // source Z up -> Three.js Y up.
  return new THREE.Vector3(
    -num(p.y),
    num(p.z),
    -num(p.x)
  );
}

function componentPoint(component, index) {
  const p = component?.points?.[index] || null;
  return p ? mapCoord(p) : null;
}

function componentCoord(component) {
  const p = component?.coOrds || component?.coord || component?.points?.[0] || null;
  return p ? mapCoord(p) : null;
}

function boreDiameter(component) {
  return Math.max(
    8,
    num(
      component?.bore ??
      component?.points?.[0]?.bore ??
      component?.points?.[1]?.bore ??
      component?.attributes?.BORE ??
      component?.attributes?.OD ??
      100
    )
  );
}

function colorForComponent(component, palette) {
  const type = String(component?.type || '').toUpperCase();

  if (type === 'PIPE' || type === 'STEEL') return palette.pipe;
  if (type === 'SUPPORT' || type === 'ANCI') return palette.support;
  if (type.includes('REDUCER')) return palette.reducer;
  if (type === 'UNKNOWN') return palette.unknown;

  return palette.fitting;
}

function makeMaterial(color, opacity) {
  return new THREE.MeshStandardMaterial({
    color,
    transparent: opacity < 1,
    opacity,
    roughness: 0.36,
    metalness: 0.12,
    depthWrite: opacity >= 1,
  });
}

function createCylinder(startVec, endVec, radius, material) {
  const diff = new THREE.Vector3().subVectors(endVec, startVec);
  const length = diff.length();

  if (length < 0.1) return null;

  const mid = new THREE.Vector3().addVectors(startVec, endVec).multiplyScalar(0.5);
  const quat = new THREE.Quaternion().setFromUnitVectors(
    new THREE.Vector3(0, 1, 0),
    diff.clone().normalize()
  );

  const geo = new THREE.CylinderGeometry(radius, radius, length, 16);
  const mesh = new THREE.Mesh(geo, material);

  mesh.position.copy(mid);
  mesh.quaternion.copy(quat);

  return mesh;
}

function createSphere(pos, radius, material) {
  const geo = new THREE.SphereGeometry(radius, 16, 12);
  const mesh = new THREE.Mesh(geo, material);
  mesh.position.copy(pos);
  return mesh;
}

function createBox(pos, size, material) {
  const geo = new THREE.BoxGeometry(size, size, size);
  const mesh = new THREE.Mesh(geo, material);
  mesh.position.copy(pos);
  return mesh;
}

function createComponentMesh(component, dataset) {
  const palette = dataset.palette;
  const color = colorForComponent(component, palette);
  const opacity = Number(palette.opacity ?? 1);
  const material = makeMaterial(color, opacity);
  const type = String(component?.type || '').toUpperCase();

  const p1 = componentPoint(component, 0);
  const p2 = componentPoint(component, 1);
  const coord = componentCoord(component);
  const radius = Math.max(2, boreDiameter(component) / 2);

  let mesh = null;

  if (p1 && p2) {
    mesh = createCylinder(p1, p2, radius, material);
  } else if (coord && (type === 'SUPPORT' || type === 'ANCI')) {
    mesh = createBox(coord, Math.max(20, radius * 1.2), material);
  } else if (coord) {
    mesh = createSphere(coord, Math.max(8, radius), material);
  }

  if (!mesh) return null;

  mesh.userData = {
    ...(mesh.userData || {}),
    selectable: true,
    compareDatasetId: dataset.id,
    compareComponentId: component.id,
    component,
    componentType: type,
  };

  return mesh;
}

function disposeObject(obj) {
  if (!obj) return;

  obj.traverse?.((child) => {
    if (child.geometry) child.geometry.dispose?.();

    const materials = Array.isArray(child.material)
      ? child.material
      : child.material ? [child.material] : [];

    for (const mat of materials) {
      mat.dispose?.();
    }
  });
}

export class XmlCompareRenderer {
  constructor(container) {
    this.container = container;
    this.scene = null;
    this.camera = null;
    this.renderer = null;
    this.controls = null;
    this.datasetGroups = new Map();
    this._animId = 0;
    this._onResize = null;

    this.init();
  }

  init() {
    const width = this.container.clientWidth || 900;
    const height = this.container.clientHeight || 620;

    this.scene = new THREE.Scene();
    this.scene.background = new THREE.Color(0x0b1220);

    this.camera = new THREE.PerspectiveCamera(55, width / height, 0.1, 10000000);
    this.camera.position.set(6000, 5000, 6000);
    this.camera.lookAt(0, 0, 0);

    this.renderer = new THREE.WebGLRenderer({ antialias: true, alpha: false });
    this.renderer.setSize(width, height);
    this.renderer.setPixelRatio(window.devicePixelRatio || 1);

    this.container.innerHTML = '';
    this.container.appendChild(this.renderer.domElement);

    this.controls = new OrbitControls(this.camera, this.renderer.domElement);
    this.controls.enableDamping = true;

    this.scene.add(new THREE.AmbientLight(0xffffff, 0.72));

    const dir = new THREE.DirectionalLight(0xffffff, 0.85);
    dir.position.set(3000, 6000, 4000);
    this.scene.add(dir);

    const grid = new THREE.GridHelper(10000, 40, 0x2f3b52, 0x1d2a3f);
    grid.name = 'xml-compare-grid';
    this.scene.add(grid);

    const axes = new THREE.AxesHelper(1200);
    axes.name = 'xml-compare-axes';
    this.scene.add(axes);

    this._onResize = () => this.resize();
    window.addEventListener('resize', this._onResize);

    this.animate();
  }

  animate() {
    this._animId = requestAnimationFrame(() => this.animate());

    this.controls?.update();
    this.renderer?.render(this.scene, this.camera);
  }

  dispose() {
    if (this._animId) cancelAnimationFrame(this._animId);
    if (this._onResize) window.removeEventListener('resize', this._onResize);

    for (const group of this.datasetGroups.values()) {
      disposeObject(group);
      this.scene.remove(group);
    }

    this.datasetGroups.clear();

    if (this.renderer) {
      this.renderer.dispose?.();
      this.renderer.domElement?.remove?.();
    }

    this.controls?.dispose?.();
  }

  resize() {
    const width = this.container.clientWidth || 900;
    const height = this.container.clientHeight || 620;

    this.camera.aspect = width / height;
    this.camera.updateProjectionMatrix();

    this.renderer.setSize(width, height);
  }

  setDatasets(datasets = []) {
    for (const group of this.datasetGroups.values()) {
      this.scene.remove(group);
      disposeObject(group);
    }

    this.datasetGroups.clear();

    for (const dataset of datasets) {
      if (!dataset || dataset.visible === false) continue;

      const group = this._buildDatasetGroup(dataset);

      this.datasetGroups.set(dataset.id, group);
      this.scene.add(group);
    }

    this.fitAll();
  }

  _buildDatasetGroup(dataset) {
    const group = new THREE.Group();
    group.name = `xml-compare-dataset-${dataset.id}`;
    group.userData.compareDatasetId = dataset.id;

    const offset = dataset.transform?.offset || { x: 0, y: 0, z: 0 };
    group.position.set(num(offset.x), num(offset.z), -num(offset.y));

    for (const component of dataset.components || []) {
      if (component.compareVisible === false) continue;

      const mesh = createComponentMesh(component, dataset);
      if (!mesh) continue;

      group.add(mesh);
    }

    return group;
  }

  setDatasetVisibility(datasetId, visible) {
    const group = this.datasetGroups.get(datasetId);
    if (group) group.visible = !!visible;
  }

  fitAll() {
    const box = new THREE.Box3();

    for (const group of this.datasetGroups.values()) {
      if (!group.visible) continue;
      box.expandByObject(group);
    }

    if (box.isEmpty()) return;

    const center = box.getCenter(new THREE.Vector3());
    const size = box.getSize(new THREE.Vector3());
    const maxDim = Math.max(size.x, size.y, size.z, 1);
    const distance = maxDim * 1.65;

    this.controls.target.copy(center);
    this.camera.position.set(
      center.x + distance,
      center.y + distance * 0.75,
      center.z + distance
    );

    this.camera.near = Math.max(0.1, maxDim / 1000);
    this.camera.far = Math.max(100000, maxDim * 40);
    this.camera.updateProjectionMatrix();

    this.controls.update();
  }
}
```

```js
async function handleImport(container, datasetId, file) {
  try {
    notify({
      type: 'info',
      message: `Importing XML ${datasetId}: ${file.name}`,
    });

    const dataset = await importXmlCompareDataset(file, datasetId);
    dataset.visible = true;

    _compareState.datasets[datasetId] = dataset;

    const toggle = container.querySelector(`[data-xml-compare-visible="${datasetId}"]`);
    if (toggle) toggle.checked = true;

    renderDatasets(container);

    notify({
      type: 'info',
      message: `XML ${datasetId} loaded: ${dataset.components.length} component(s).`,
    });
  } catch (err) {

function bindXmlCompareTab(container) {
  if (container.dataset.xmlCompareBound === 'true') return;
  container.dataset.xmlCompareBound = 'true';

  container.addEventListener('change', async (event) => {
    const importInput = event.target.closest('[data-xml-compare-import]');

    if (importInput && container.contains(importInput)) {
      const datasetId = importInput.dataset.xmlCompareImport;
      const file = importInput.files?.[0] || null;

      importInput.value = '';

      if (file) await handleImport(container, datasetId, file);
      return;
    }

    const visibleInput = event.target.closest('[data-xml-compare-visible]');

    if (visibleInput && container.contains(visibleInput)) {
      const datasetId = visibleInput.dataset.xmlCompareVisible;
      _compareState.datasets[datasetId].visible = !!visibleInput.checked;

      if (_renderer) {
        _renderer.setDatasetVisibility(datasetId, !!visibleInput.checked);
      }
    }
  });

  container.addEventListener('click', (event) => {
    const action = event.target.closest('[data-xml-compare-action]')?.dataset.xmlCompareAction;

    if (action === 'fit') {
      _renderer?.fitAll();
    }
  });
}
```

```js
export function normalizeXmlCompareOffset(offset = {}) {
  return {
    x: Number.isFinite(Number(offset.x)) ? Number(offset.x) : 0,
    y: Number.isFinite(Number(offset.y)) ? Number(offset.y) : 0,

export function setXmlCompareDatasetOffset(dataset, offset = {}) {
  if (!dataset) return dataset;

  const nextOffset = normalizeXmlCompareOffset(offset);

  dataset.transform = {

  return dataset;
}

}
```

## 2D. Add transform and pick methods inside `XmlCompareRenderer`

Add these methods before `fitAll()`:

```js
  setDatasetTransform(datasetId, transform = {}) {
    const group = this.datasetGroups.get(datasetId);
    if (!group) return false;

    const offset = transform.offset || { x: 0, y: 0, z: 0 };

    group.position.copy(threeOffsetFromSourceOffset(offset));
    group.userData.compareTransform = transform;

    return true;
  }

  setDatasetOffset(datasetId, offset = {}) {
    return this.setDatasetTransform(datasetId, {
      offset,
    });
  }

  pickComponentPointFromEvent(event, datasetId = '') {
    if (!this.renderer || !this.camera || !this.scene) return null;

    const rect = this.renderer.domElement.getBoundingClientRect();

    if (!rect.width || !rect.height) return null;

    this.pointer.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
    this.pointer.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;

    this.raycaster.setFromCamera(this.pointer, this.camera);

    const roots = datasetId
      ? [this.datasetGroups.get(datasetId)].filter(Boolean)
      : [...this.datasetGroups.values()];

    const objects = [];

    for (const root of roots) {
      root.traverse((child) => {
        if (child?.isMesh && child.userData?.selectable) {
          objects.push(child);
        }
      });
    }

    const hits = this.raycaster.intersectObjects(objects, false);

    if (!hits.length) return null;

    const hit = hits[0];
    const object = hit.object;

    return {
      point: hit.point.clone(),
      datasetId: object.userData.compareDatasetId || datasetId,
      componentId: object.userData.compareComponentId || '',
      componentType: object.userData.componentType || '',
      component: object.userData.component || null,
      object,
    };
  }

  offsetForPickedPointToOrigin(pick) {
    if (!pick?.point || !pick.datasetId) return null;

    const group = this.datasetGroups.get(pick.datasetId);
    if (!group) return null;

    // pickedWorld = localPoint + currentGroupPosition.
    // To move picked point to world origin:
    // newGroupPosition = currentGroupPosition - pickedWorld.
    const newThreeOffset = group.position.clone().sub(pick.point);

    return sourceOffsetFromThreeOffset(newThreeOffset);
  }
```

```js
function applyDatasetOffset(container, datasetId, offset, options = {}) {
  const dataset = _compareState.datasets[datasetId];

  if (!dataset || !dataset.components?.length) {
    notify({
      type: 'warning',
      message: `XML ${datasetId} is not loaded.`,
    });
    return false;
  }

  setXmlCompareDatasetOffset(dataset, offset);

  if (_renderer) {
    _renderer.setDatasetOffset(datasetId, dataset.transform.offset);
  }

  const target = container.querySelector('[data-xml-compare-origin-target]');
  if (target) target.value = datasetId;

  setOffsetInputs(container, datasetId);

  if (options.fit !== false) {
    _renderer?.fitAll();
  }

  if (options.notifyUser) {
    notify({
      type: 'info',
      message: `XML ${datasetId} offset set to X=${dataset.transform.offset.x}, Y=${dataset.transform.offset.y}, Z=${dataset.transform.offset.z}.`,
    });
  }

  updateSummary(container);

  return true;
}

function resetDatasetOrigin(container, datasetId) {
  const dataset = _compareState.datasets[datasetId];

  if (!dataset) return;

  resetXmlCompareDatasetTransform(dataset);

  if (_renderer) {
    _renderer.setDatasetOffset(datasetId, dataset.transform.offset);
  }

  setOffsetInputs(container, datasetId);
  setOriginStatus(container, `XML ${datasetId} origin reset.`);
  _renderer?.fitAll();

  notify({
    type: 'info',
    message: `XML ${datasetId} origin reset.`,
  });
}

function startOriginPick(container, datasetId) {
  if (!_compareState.datasets[datasetId]?.components?.length) {
    notify({
      type: 'warning',
      message: `Load XML ${datasetId} before using Pick Origin ${datasetId}.`,
    });
    return;
  }

  _originPickDatasetId = datasetId;

  const canvas = container.querySelector('[data-xml-compare-canvas]');
  if (canvas) {
    canvas.classList.add('is-picking-origin');
  }

  setOriginStatus(container, `Pick a point/component on XML ${datasetId}. That point will move to global origin.`);
}

function stopOriginPick(container) {
  _originPickDatasetId = '';

  const canvas = container.querySelector('[data-xml-compare-canvas]');
  if (canvas) {
    canvas.classList.remove('is-picking-origin');
  }

  setOriginStatus(container, 'Origin tool idle');
}

function handleOriginPickPointer(container, event) {
  if (!_originPickDatasetId || !_renderer) return;

  const datasetId = _originPickDatasetId;
  const pick = _renderer.pickComponentPointFromEvent(event, datasetId);

  if (!pick) {
    notify({
      type: 'warning',
      message: `No XML ${datasetId} component picked. Try clicking directly on geometry.`,
    });
    return;
  }

  const offset = _renderer.offsetForPickedPointToOrigin(pick);

  if (!offset) {
    notify({
      type: 'error',
      message: `Could not compute origin offset for XML ${datasetId}.`,
    });
    return;
  }

  applyDatasetOffset(container, datasetId, offset, {
    notifyUser: true,
    fit: true,
  });

  setOriginStatus(
    container,
    `XML ${datasetId} origin picked from ${pick.componentType || 'component'} ${pick.componentId || ''}.`
  );

  stopOriginPick(container);
}
```

```js
/**
 * XmlCompareTree.js
 *
 * X4:
 * - Build hierarchy tree for XML A / XML B.
 * - Group by dataset -> pipeline -> type -> component.
 * - Support multiselect and visibility by component/group.
 */

function clean(value) {
  return String(value ?? '').trim();
}

function componentType(component) {
  return clean(component?.type || 'UNKNOWN').toUpperCase() || 'UNKNOWN';
}

function componentPipeline(component) {
  const attrs = component?.attributes || {};
  const source = component?.source || {};

  return clean(
    attrs['PIPELINE-REFERENCE'] ||
    attrs.PIPELINE_REFERENCE ||
    attrs.LINE_NO ||
    attrs.LINE ||
    component?.lineNo ||
    source.lineNo ||
    source.pipelineRef ||
    component?.pipelineRef ||
    'NO PIPELINE'
  );
}

function componentLabel(component, index) {
  const attrs = component?.attributes || {};
  const ref =
    component?.compareOriginalId ||
    component?.refNo ||
    attrs['COMPONENT-ATTRIBUTE97'] ||
    attrs.COMPONENT_ATTRIBUTE97 ||
    attrs['COMPONENT-IDENTIFIER'] ||
    component?.id ||
    `#${index + 1}`;

  return `${componentType(component)} · ${clean(ref)}`;
}

function makeNode({
  id,
  parentId = '',
  datasetId = '',
  type = 'group',
  label = '',
  componentId = '',
  componentIds = [],
  children = [],
}) {
  return {
    id,
    parentId,
    datasetId,
    type,
    label,
    componentId,
    componentIds,
    children,
  };
}

export function buildXmlCompareTree(datasets = []) {
  const roots = [];

  for (const dataset of datasets || []) {
    if (!dataset) continue;

    const datasetId = dataset.id;
    const datasetNode = makeNode({
      id: `dataset:${datasetId}`,
      datasetId,
      type: 'dataset',
      label: `XML ${datasetId}${dataset.sourceFileName ? ` — ${dataset.sourceFileName}` : ''}`,
    });

    const pipelineMap = new Map();

    (dataset.components || []).forEach((component, index) => {
      const pipeline = componentPipeline(component);
      const type = componentType(component);
      const componentId = component.id;

      if (!pipelineMap.has(pipeline)) {
        const node = makeNode({
          id: `pipeline:${datasetId}:${pipeline}`,
          parentId: datasetNode.id,
          datasetId,
          type: 'pipeline',
          label: pipeline,
        });

        pipelineMap.set(pipeline, {
          node,
          typeMap: new Map(),
        });

        datasetNode.children.push(node);
      }

      const pipelineBucket = pipelineMap.get(pipeline);

      if (!pipelineBucket.typeMap.has(type)) {
        const typeNode = makeNode({
          id: `type:${datasetId}:${pipeline}:${type}`,
          parentId: pipelineBucket.node.id,
          datasetId,
          type: 'type',
          label: type,
        });

        pipelineBucket.typeMap.set(type, typeNode);
        pipelineBucket.node.children.push(typeNode);
      }

      const typeNode = pipelineBucket.typeMap.get(type);
      const componentNode = makeNode({
        id: `component:${componentId}`,
        parentId: typeNode.id,
        datasetId,
        type: 'component',
        label: componentLabel(component, index),
        componentId,
        componentIds: [componentId],
      });

      typeNode.children.push(componentNode);
    });

    roots.push(datasetNode);
  }

  annotateComponentIds(roots);
  return roots;
}

export function flattenXmlCompareTree(nodes = []) {
  const out = [];

  function walk(node, depth = 0) {
    out.push({ node, depth });
    for (const child of node.children || []) {
      walk(child, depth + 1);
    }
  }

  for (const node of nodes || []) {
    walk(node, 0);
  }

  return out;
}

export function collectTreeComponentIds(node) {
  if (!node) return [];

  if (node.componentId) return [node.componentId];

  const ids = [];
  for (const child of node.children || []) {
    ids.push(...collectTreeComponentIds(child));
  }

  return [...new Set(ids)];
}

function annotateComponentIds(nodes = []) {
  for (const node of nodes) {
    node.componentIds = collectTreeComponentIds(node);
    annotateComponentIds(node.children || []);
  }
}

export function findTreeNodeById(nodes = [], nodeId) {
  for (const node of nodes || []) {
    if (node.id === nodeId) return node;

    const child = findTreeNodeById(node.children || [], nodeId);
    if (child) return child;
  }

  return null;
}

export function findTreeNodeForComponent(nodes = [], componentId) {
  return findTreeNodeById(nodes, `component:${componentId}`);
}

export function toggleSelectionSet(selectionSet, ids = [], mode = 'replace') {
  const next = new Set(mode === 'replace' ? [] : selectionSet || []);

  if (mode === 'toggle') {
    for (const id of ids) {
      if (next.has(id)) next.delete(id);
      else next.add(id);
    }

    return next;
  }

  if (mode === 'add') {
    for (const id of ids) next.add(id);
    return next;
  }

  for (const id of ids) next.add(id);
  return next;
}

export function setVisibilityForIds(hiddenSet, ids = [], visible = true) {
  const next = new Set(hiddenSet || []);

  for (const id of ids) {
    if (visible) next.delete(id);
    else next.add(id);
  }

  return next;
}
```

```js
function refreshXmlCompareTreeModel() {
  _treeNodes = buildXmlCompareTree(loadedDatasets());
  _flatTree = flattenXmlCompareTree(_treeNodes);
}

function renderXmlCompareTree(container) {
  refreshXmlCompareTreeModel();

  const host = container.querySelector('[data-xml-compare-tree]');
  if (!host) return;

  if (!_flatTree.length) {
    host.innerHTML = '<div class="xml-compare-tree-empty">Import XML A / XML B to build hierarchy.</div>';
    return;
  }

  host.innerHTML = _flatTree.map(({ node, depth }, index) => {
    const ids = collectTreeComponentIds(node);
    const selected = ids.some(id => _selectedComponentIds.has(id));
    const allHidden = ids.length > 0 && ids.every(id => _hiddenComponentIds.has(id));
    const partiallyHidden = ids.some(id => _hiddenComponentIds.has(id)) && !allHidden;

    return `
      <div
        class="xml-compare-tree-row ${selected ? 'is-selected' : ''} ${allHidden ? 'is-hidden' : ''}"
        data-tree-node-id="${esc(node.id)}"
        data-tree-index="${index}"
        style="--tree-depth:${depth};"
        title="${esc(node.label)}"
      >
        <input
          type="checkbox"
          data-tree-visible="${esc(node.id)}"
          ${allHidden ? '' : 'checked'}
          ${partiallyHidden ? 'data-partial="true"' : ''}
          title="Toggle visibility"
        >
        <span class="tree-node-kind">${esc(node.type)}</span>
        <span class="tree-node-label">${esc(node.label)}</span>
        <span class="tree-node-count">${ids.length || ''}</span>
      </div>
    `;
  }).join('');

  host.querySelectorAll('[data-partial="true"]').forEach(input => {
    input.indeterminate = true;
  });
}

function applySelectionToRenderer(container) {
  if (_renderer) {
    _renderer.setSelectedComponentIds([..._selectedComponentIds]);
  }

  renderXmlCompareTree(container);
}

function applyVisibilityToRenderer(container) {
  if (_renderer) {
    _renderer.setHiddenComponentIds([..._hiddenComponentIds]);
  }

  renderXmlCompareTree(container);
}

function selectComponentIds(container, ids, mode = 'replace') {
  _selectedComponentIds = toggleSelectionSet(_selectedComponentIds, ids, mode);
  applySelectionToRenderer(container);
}

function selectTreeNode(container, node, event = {}) {
  if (!node) return;

  const ids = collectTreeComponentIds(node);
  const index = Number(event.currentTarget?.dataset?.treeIndex ?? -1);

  if (event.shiftKey && _lastTreeNodeIndex >= 0 && index >= 0) {
    const start = Math.min(_lastTreeNodeIndex, index);
    const end = Math.max(_lastTreeNodeIndex, index);
    const rangeIds = [];

    for (let i = start; i <= end; i += 1) {
      rangeIds.push(...collectTreeComponentIds(_flatTree[i]?.node));
    }

    selectComponentIds(container, rangeIds, 'replace');
  } else if (event.ctrlKey || event.metaKey) {
    selectComponentIds(container, ids, 'toggle');
  } else {
    selectComponentIds(container, ids, 'replace');
  }

  _lastTreeNodeIndex = index;
}

function syncTreeToCanvasPick(container, pick, event = {}) {
  if (!pick?.componentId) return;

  const node = findTreeNodeForComponent(_treeNodes, pick.componentId);
  if (!node) return;

  selectComponentIds(
    container,
    [pick.componentId],
    event.ctrlKey || event.metaKey ? 'toggle' : 'replace'
  );

  const row = container.querySelector(`[data-tree-node-id="${CSS.escape(node.id)}"]`);
  if (row) {
    row.scrollIntoView({ block: 'nearest' });
  }
}

function toggleTreeVisibility(container, node, visible) {
  if (!node) return;

  const ids = collectTreeComponentIds(node);
  _hiddenComponentIds = setVisibilityForIds(_hiddenComponentIds, ids, visible);
  applyVisibilityToRenderer(container);
}

function hideSelected(container) {
  _hiddenComponentIds = setVisibilityForIds(_hiddenComponentIds, [..._selectedComponentIds], false);
  applyVisibilityToRenderer(container);
}

function isolateSelected(container) {
  const selected = new Set(_selectedComponentIds);
  const allIds = [];

  for (const dataset of loadedDatasets()) {
    for (const component of dataset.components || []) {
      allIds.push(component.id);
    }
  }

  _hiddenComponentIds = new Set(allIds.filter(id => !selected.has(id)));
  applyVisibilityToRenderer(container);
}

function showAllComponents(container) {
  _hiddenComponentIds = new Set();

  if (_renderer) {
    _renderer.showAllComponents();
  }

  applyVisibilityToRenderer(container);
}
```

```js
export const XML_COMPARE_VIEW_MODES = Object.freeze({
  OVERLAY: 'OVERLAY',
  A_ONLY: 'A_ONLY',
  B_ONLY: 'B_ONLY',
  GHOST: 'GHOST',
  CHANGED: 'CHANGED',
});

const XML_COMPARE_STORAGE_KEY = 'pcf_glb_xml_compare_state_v1';

export function normalizeXmlCompareViewMode(value) {
  const v = String(value || '').trim().toUpperCase();

  if (v === XML_COMPARE_VIEW_MODES.A_ONLY) return XML_COMPARE_VIEW_MODES.A_ONLY;
  if (v === XML_COMPARE_VIEW_MODES.B_ONLY) return XML_COMPARE_VIEW_MODES.B_ONLY;
  if (v === XML_COMPARE_VIEW_MODES.GHOST) return XML_COMPARE_VIEW_MODES.GHOST;
  if (v === XML_COMPARE_VIEW_MODES.CHANGED) return XML_COMPARE_VIEW_MODES.CHANGED;

  return XML_COMPARE_VIEW_MODES.OVERLAY;
}

export function visibleDatasetIdsForViewMode(mode) {
  const normalized = normalizeXmlCompareViewMode(mode);

  if (normalized === XML_COMPARE_VIEW_MODES.A_ONLY) return ['A'];
  if (normalized === XML_COMPARE_VIEW_MODES.B_ONLY) return ['B'];

  return ['A', 'B'];
}

export function xmlCompareComponentSignature(component) {
  const attrs = component?.attributes || {};
  const source = component?.source || {};

  const pipeline = clean(
    attrs['PIPELINE-REFERENCE'] ||
    attrs.PIPELINE_REFERENCE ||
    attrs.LINE_NO ||
    source.lineNo ||
    component?.pipelineRef ||
    ''
  );

  const type = clean(component?.type || 'UNKNOWN').toUpperCase();

  const p1 = component?.points?.[0] || component?.coOrds || {};
  const p2 = component?.points?.[1] || {};

  const coordKey = [
    Number(p1.x || 0).toFixed(1),
    Number(p1.y || 0).toFixed(1),
    Number(p1.z || 0).toFixed(1),
    Number(p2.x || 0).toFixed(1),
    Number(p2.y || 0).toFixed(1),
    Number(p2.z || 0).toFixed(1),
  ].join(',');

  return `${pipeline}::${type}::${coordKey}`;
}

export function classifyXmlCompareChanges(datasetA, datasetB) {
  const aComponents = Array.isArray(datasetA?.components) ? datasetA.components : [];
  const bComponents = Array.isArray(datasetB?.components) ? datasetB.components : [];

  const aSet = new Set(aComponents.map(xmlCompareComponentSignature));
  const bSet = new Set(bComponents.map(xmlCompareComponentSignature));

  const changedIds = new Set();

  for (const component of aComponents) {
    if (!bSet.has(xmlCompareComponentSignature(component))) {
      changedIds.add(component.id);
    }
  }

  for (const component of bComponents) {
    if (!aSet.has(xmlCompareComponentSignature(component))) {
      changedIds.add(component.id);
    }
  }

  return changedIds;
}

export function serializeXmlCompareUiState(compareState = {}, extra = {}) {
  const datasets = compareState.datasets || {};

  return {
    version: 1,
    viewMode: normalizeXmlCompareViewMode(extra.viewMode),
    hiddenComponentIds: Array.from(extra.hiddenComponentIds || []),
    selectedComponentIds: Array.from(extra.selectedComponentIds || []),
    datasets: {
      A: {
        visible: datasets.A?.visible !== false,
        transform: datasets.A?.transform || {
          offset: { x: 0, y: 0, z: 0 },
          origin: { x: 0, y: 0, z: 0 },
          scale: 1,
        },
      },
      B: {
        visible: datasets.B?.visible !== false,
        transform: datasets.B?.transform || {
          offset: { x: 0, y: 0, z: 0 },
          origin: { x: 0, y: 0, z: 0 },
          scale: 1,
        },
      },
    },
  };
}

export function saveXmlCompareUiState(compareState = {}, extra = {}) {
  const payload = serializeXmlCompareUiState(compareState, extra);

  try {
    localStorage.setItem(XML_COMPARE_STORAGE_KEY, JSON.stringify(payload));
  } catch {
    // localStorage may be unavailable in tests/private mode
  }

  return payload;
}

export function loadXmlCompareUiState() {
  try {
    const parsed = JSON.parse(localStorage.getItem(XML_COMPARE_STORAGE_KEY) || '{}');

    return {
      version: 1,
      viewMode: normalizeXmlCompareViewMode(parsed.viewMode),
      hiddenComponentIds: Array.isArray(parsed.hiddenComponentIds) ? parsed.hiddenComponentIds : [],
      selectedComponentIds: Array.isArray(parsed.selectedComponentIds) ? parsed.selectedComponentIds : [],
      datasets: {
        A: {
          visible: parsed.datasets?.A?.visible !== false,
          transform: parsed.datasets?.A?.transform || {
            offset: { x: 0, y: 0, z: 0 },
            origin: { x: 0, y: 0, z: 0 },
            scale: 1,
          },
        },
        B: {
          visible: parsed.datasets?.B?.visible !== false,
          transform: parsed.datasets?.B?.transform || {
            offset: { x: 0, y: 0, z: 0 },
            origin: { x: 0, y: 0, z: 0 },
            scale: 1,
          },
        },
      },
    };
  } catch {
    return {
      version: 1,
      viewMode: XML_COMPARE_VIEW_MODES.OVERLAY,
      hiddenComponentIds: [],
      selectedComponentIds: [],
      datasets: {
        A: {
          visible: true,
          transform: { offset: { x: 0, y: 0, z: 0 }, origin: { x: 0, y: 0, z: 0 }, scale: 1 },
        },
        B: {
          visible: true,
          transform: { offset: { x: 0, y: 0, z: 0 }, origin: { x: 0, y: 0, z: 0 }, scale: 1 },
        },
      },
    };
  }
}
```

```js
function setMeshOpacity(mesh, opacity) {
  if (!mesh?.material) return;

  const materials = Array.isArray(mesh.material) ? mesh.material : [mesh.material];

  for (const material of materials) {
    material.transparent = opacity < 1;
    material.opacity = opacity;
    material.depthWrite = opacity >= 1;
    material.needsUpdate = true;
  }
}

function setMeshChangedColor(mesh, changed) {
  if (!mesh?.material || !changed) return;

  const materials = Array.isArray(mesh.material) ? mesh.material : [mesh.material];

  for (const material of materials) {
    if (material.color) material.color.setHex(0xff1744);
    if (material.emissive) material.emissive.setHex(0x661111);
  }
}
```

## 2D. Add these methods inside class

```js
  setViewMode(mode, datasets = []) {
    this.viewMode = normalizeXmlCompareViewMode(mode);
    this.changedComponentIds = this.viewMode === XML_COMPARE_VIEW_MODES.CHANGED
      ? classifyXmlCompareChanges(datasets[0], datasets[1])
      : new Set();

    this._applyViewMode();

  _applyViewMode() {
    const visibleDatasetIds = new Set(visibleDatasetIdsForViewMode(this.viewMode));

    for (const [datasetId, group] of this.datasetGroups.entries()) {
      group.visible = visibleDatasetIds.has(datasetId);

      group.traverse((child) => {
        if (!child?.isMesh || !child.userData?.selectable) return;

        const componentId = child.userData.compareComponentId;
        const hiddenByTree = this.hiddenComponentIds.has(componentId);

        child.visible = !hiddenByTree;

        if (this.viewMode === XML_COMPARE_VIEW_MODES.GHOST) {
          setMeshOpacity(child, datasetId === 'A' ? 0.35 : 0.72);
        } else {
          setMeshOpacity(child, datasetId === 'B' ? 0.72 : 1.0);
        }

        if (this.viewMode === XML_COMPARE_VIEW_MODES.CHANGED) {
          setMeshChangedColor(child, this.changedComponentIds.has(componentId));
        }
      });
    }
  }
```

```js
container.addEventListener('pointerdown', (event) => {
  const canvas = event.target.closest('[data-xml-compare-canvas]');
  if (!canvas || !container.contains(canvas)) return;

  _pointerDownInfo = {
    x: event.clientX,
    y: event.clientY,
    time: performance.now(),
    button: event.button,
    ctrlKey: event.ctrlKey,
    metaKey: event.metaKey,
    shiftKey: event.shiftKey,
  };
});

container.addEventListener('pointerup', (event) => {
  const canvas = event.target.closest('[data-xml-compare-canvas]');
  if (!canvas || !container.contains(canvas)) return;
  if (!_pointerDownInfo) return;

  const dx = event.clientX - _pointerDownInfo.x;
  const dy = event.clientY - _pointerDownInfo.y;
  const dist = Math.sqrt(dx * dx + dy * dy);
  const wasClick = dist <= XML_COMPARE_CLICK_DISTANCE_PX && _pointerDownInfo.button === 0;

  const down = _pointerDownInfo;
  _pointerDownInfo = null;

  if (!wasClick) {
    return;
  }

  if (_originPickDatasetId) {
    event.preventDefault();
    handleOriginPickPointer(container, event);
    return;
  }

  const pick = _renderer?.pickComponentFromEvent(event);
  if (!pick?.componentId) return;

  event.preventDefault();

  syncTreeToCanvasPick(container, pick, {
    ctrlKey: down.ctrlKey || event.ctrlKey,
    metaKey: down.metaKey || event.metaKey,
    shiftKey: down.shiftKey || event.shiftKey,
  });
```

```js
function componentComparePipeline(component) {
  const attrs = component?.attributes || {};
  const source = component?.source || {};

  return clean(
    attrs['PIPELINE-REFERENCE'] ||
    attrs.PIPELINE_REFERENCE ||
    attrs.LINE_NO ||
    source.lineNo ||
    component?.pipelineRef ||
    'NO PIPELINE'
  );
}

function componentCompareType(component) {
  return clean(component?.type || 'UNKNOWN').toUpperCase() || 'UNKNOWN';
}

function componentCompareLabel(component) {
  const attrs = component?.attributes || {};

  return clean(
    component?.compareOriginalId ||
    component?.refNo ||
    attrs['COMPONENT-ATTRIBUTE97'] ||
    attrs.COMPONENT_ATTRIBUTE97 ||
    attrs['COMPONENT-IDENTIFIER'] ||
    component?.id ||
    ''
  );
}

function componentCompareInfo(component, datasetId, status) {
  return {
    datasetId,
    status,
    componentId: component?.id || '',
    type: componentCompareType(component),
    pipeline: componentComparePipeline(component),
    label: componentCompareLabel(component),
    signature: xmlCompareComponentSignature(component),
  };
}

function groupComponentsBySignature(components = []) {
  const map = new Map();

  for (const component of components || []) {
    const signature = xmlCompareComponentSignature(component);

    if (!map.has(signature)) map.set(signature, []);
    map.get(signature).push(component);
  }

  return map;
}

export function buildXmlCompareDiffReport(datasetA, datasetB) {
  const aComponents = Array.isArray(datasetA?.components) ? datasetA.components : [];
  const bComponents = Array.isArray(datasetB?.components) ? datasetB.components : [];

  const aBySig = groupComponentsBySignature(aComponents);
  const bBySig = groupComponentsBySignature(bComponents);

  const signatures = new Set([
    ...aBySig.keys(),
    ...bBySig.keys(),
  ]);

  const rows = [];
  const matched = [];
  const onlyA = [];
  const onlyB = [];

  for (const signature of signatures) {
    const aList = [...(aBySig.get(signature) || [])];
    const bList = [...(bBySig.get(signature) || [])];

    const pairCount = Math.min(aList.length, bList.length);

    for (let i = 0; i < pairCount; i += 1) {
      const a = componentCompareInfo(aList[i], 'A', 'MATCHED');
      const b = componentCompareInfo(bList[i], 'B', 'MATCHED');

      const row = {
        status: 'MATCHED',
        signature,
        a,
        b,
      };

      rows.push(row);
      matched.push(row);
    }

    for (let i = pairCount; i < aList.length; i += 1) {
      const row = {
        status: 'ONLY_A',
        signature,
        a: componentCompareInfo(aList[i], 'A', 'ONLY_A'),
        b: null,
      };

      rows.push(row);
      onlyA.push(row);
    }

    for (let i = pairCount; i < bList.length; i += 1) {
      const row = {
        status: 'ONLY_B',
        signature,
        a: null,
        b: componentCompareInfo(bList[i], 'B', 'ONLY_B'),
      };

      rows.push(row);
      onlyB.push(row);
    }
  }

  rows.sort((r1, r2) => {
    const order = { ONLY_A: 0, ONLY_B: 1, MATCHED: 2 };
    const a = order[r1.status] ?? 9;
    const b = order[r2.status] ?? 9;

    if (a !== b) return a - b;

    const p1 = r1.a?.pipeline || r1.b?.pipeline || '';
    const p2 = r2.a?.pipeline || r2.b?.pipeline || '';

    return String(p1).localeCompare(String(p2));
  });

  return {
    schemaVersion: 'xml-compare-report/v1',
    generatedAt: new Date().toISOString(),
    files: {
      A: datasetA?.sourceFileName || '',
      B: datasetB?.sourceFileName || '',
    },
    counts: {
      totalA: aComponents.length,
      totalB: bComponents.length,
      matched: matched.length,
      onlyA: onlyA.length,
      onlyB: onlyB.length,
      changed: onlyA.length + onlyB.length,
    },
    rows,
  };
}

export function serializeXmlCompareReportJson(report) {
  return JSON.stringify(report || {}, null, 2);
}

function csvCell(value) {
  const text = String(value ?? '');
  if (/[",\n\r]/.test(text)) {
    return `"${text.replaceAll('"', '""')}"`;
  }
  return text;
}

export function serializeXmlCompareReportCsv(report) {
  const rows = Array.isArray(report?.rows) ? report.rows : [];

  const header = [
    'Status',
    'A Component Id',
    'A Type',
    'A Pipeline',
    'A Label',
    'B Component Id',
    'B Type',
    'B Pipeline',
    'B Label',
    'Signature',
  ];

  const lines = [header.map(csvCell).join(',')];

  for (const row of rows) {
    lines.push([
      row.status,
      row.a?.componentId || '',
      row.a?.type || '',
      row.a?.pipeline || '',
      row.a?.label || '',
      row.b?.componentId || '',
      row.b?.type || '',
      row.b?.pipeline || '',
      row.b?.label || '',
      row.signature || '',
    ].map(csvCell).join(','));
  }

  return `${lines.join('\n')}\n`;
}
```

# X8 — tolerance-based geometry compare

Current X7 compare is too strict because it uses exact geometry signature. X8 makes compare useful for real XML:

```text
MATCHED          same identity + within tolerance
MOVED           same identity, but endpoint/coord shifted beyond tolerance
GEOMETRY_CHANGED same identity, length/shape changed beyond tolerance
TYPE_CHANGED    same component identity but different type
ONLY_A          exists only in XML A
ONLY_B          exists only in XML B
```

```js
export const XML_COMPARE_DEFAULT_TOLERANCE = Object.freeze({
  positionMm: 5,
  lengthMm: 5,
});

export function normalizeXmlCompareTolerance(value = {}) {
  return {
    positionMm: Number.isFinite(Number(value.positionMm))
      ? Math.max(0, Number(value.positionMm))
      : XML_COMPARE_DEFAULT_TOLERANCE.positionMm,

    lengthMm: Number.isFinite(Number(value.lengthMm))
      ? Math.max(0, Number(value.lengthMm))
      : XML_COMPARE_DEFAULT_TOLERANCE.lengthMm,
  };
}

function pointOf(component, index) {
  return component?.points?.[index] || component?.coOrds || component?.coord || null;
}

function dist3(a, b) {
  if (!a || !b) return Number.POSITIVE_INFINITY;

  const dx = Number(a.x || 0) - Number(b.x || 0);
  const dy = Number(a.y || 0) - Number(b.y || 0);
  const dz = Number(a.z || 0) - Number(b.z || 0);

  return Math.sqrt(dx * dx + dy * dy + dz * dz);
}

function componentLength(component) {
  const a = pointOf(component, 0);
  const b = component?.points?.[1] || null;

  if (!a || !b) return 0;

  return dist3(a, b);
}

export function xmlCompareIdentityKey(component) {
  const pipeline = componentComparePipeline(component);
  const label = componentCompareLabel(component);
  const type = componentCompareType(component);

  // Best case: stable component/ref number exists.
  if (label) {
    return `${pipeline}::${label}`;
  }

  // Fallback: pipeline + type + coarse start-point bucket.
  const p = pointOf(component, 0) || {};
  const bucket = [
    Math.round(Number(p.x || 0) / 50),
    Math.round(Number(p.y || 0) / 50),
    Math.round(Number(p.z || 0) / 50),
  ].join(',');

  return `${pipeline}::${type}::${bucket}`;
}

export function compareXmlComponentGeometry(a, b, tolerance = {}) {
  const tol = normalizeXmlCompareTolerance(tolerance);

  if (!a || !b) {
    return {
      status: a ? 'ONLY_A' : 'ONLY_B',
      maxPointDeltaMm: Number.POSITIVE_INFINITY,
      lengthDeltaMm: Number.POSITIVE_INFINITY,
      typeChanged: false,
    };
  }

  const typeA = componentCompareType(a);
  const typeB = componentCompareType(b);

  const p1Delta = dist3(pointOf(a, 0), pointOf(b, 0));
  const p2Delta = dist3(a?.points?.[1], b?.points?.[1]);
  const maxPointDeltaMm = Math.max(
    Number.isFinite(p1Delta) ? p1Delta : 0,
    Number.isFinite(p2Delta) ? p2Delta : 0
  );

  const lengthDeltaMm = Math.abs(componentLength(a) - componentLength(b));
  const typeChanged = typeA !== typeB;

  if (typeChanged) {
    return {
      status: 'TYPE_CHANGED',
      maxPointDeltaMm,
      lengthDeltaMm,
      typeChanged,
    };
  }

  if (lengthDeltaMm > tol.lengthMm) {
    return {
      status: 'GEOMETRY_CHANGED',
      maxPointDeltaMm,
      lengthDeltaMm,
      typeChanged,
    };
  }

  if (maxPointDeltaMm > tol.positionMm) {
    return {
      status: 'MOVED',
      maxPointDeltaMm,
      lengthDeltaMm,
      typeChanged,
    };
  }

  return {
    status: 'MATCHED',
    maxPointDeltaMm,
    lengthDeltaMm,
    typeChanged,
  };
}

function groupComponentsByIdentity(components = []) {
  const map = new Map();

  for (const component of components || []) {
    const key = xmlCompareIdentityKey(component);

    if (!map.has(key)) map.set(key, []);
    map.get(key).push(component);
  }

  return map;
}

export function changedComponentIdsFromReport(report) {
  const ids = new Set();

  for (const row of report?.rows || []) {
    if (row.status === 'MATCHED') continue;

    if (row.a?.componentId) ids.add(row.a.componentId);
    if (row.b?.componentId) ids.add(row.b.componentId);
  }

  return ids;
}
```

Now **replace the existing `buildXmlCompareDiffReport(datasetA, datasetB)`** with this tolerance-aware version:

```js
export function buildXmlCompareDiffReport(datasetA, datasetB, options = {}) {
  const tolerance = normalizeXmlCompareTolerance(options.tolerance || options);

  const aComponents = Array.isArray(datasetA?.components) ? datasetA.components : [];
  const bComponents = Array.isArray(datasetB?.components) ? datasetB.components : [];

  const aByIdentity = groupComponentsByIdentity(aComponents);
  const bByIdentity = groupComponentsByIdentity(bComponents);

  const identities = new Set([
    ...aByIdentity.keys(),
    ...bByIdentity.keys(),
  ]);

  const rows = [];
  const counts = {
    totalA: aComponents.length,
    totalB: bComponents.length,
    matched: 0,
    moved: 0,
    geometryChanged: 0,
    typeChanged: 0,
    onlyA: 0,
    onlyB: 0,
    changed: 0,
  };

  for (const identity of identities) {
    const aList = [...(aByIdentity.get(identity) || [])];
    const bList = [...(bByIdentity.get(identity) || [])];

    const pairCount = Math.min(aList.length, bList.length);

    for (let i = 0; i < pairCount; i += 1) {
      const aComp = aList[i];
      const bComp = bList[i];
      const geometry = compareXmlComponentGeometry(aComp, bComp, tolerance);
      const status = geometry.status;

      const row = {
        status,
        identity,
        signature: xmlCompareComponentSignature(aComp),
        tolerance,
        geometry,
        a: componentCompareInfo(aComp, 'A', status),
        b: componentCompareInfo(bComp, 'B', status),
      };

      rows.push(row);

      if (status === 'MATCHED') counts.matched += 1;
      else if (status === 'MOVED') counts.moved += 1;
      else if (status === 'GEOMETRY_CHANGED') counts.geometryChanged += 1;
      else if (status === 'TYPE_CHANGED') counts.typeChanged += 1;
    }

    for (let i = pairCount; i < aList.length; i += 1) {
      rows.push({
        status: 'ONLY_A',
        identity,
        signature: xmlCompareComponentSignature(aList[i]),
        tolerance,
        geometry: null,
        a: componentCompareInfo(aList[i], 'A', 'ONLY_A'),
        b: null,
      });

      counts.onlyA += 1;
    }

    for (let i = pairCount; i < bList.length; i += 1) {
      rows.push({
        status: 'ONLY_B',
        identity,
        signature: xmlCompareComponentSignature(bList[i]),
        tolerance,
        geometry: null,
        a: null,
        b: componentCompareInfo(bList[i], 'B', 'ONLY_B'),
      });

      counts.onlyB += 1;
    }
  }

  counts.changed =
    counts.moved +
    counts.geometryChanged +
    counts.typeChanged +
    counts.onlyA +
    counts.onlyB;

  rows.sort((r1, r2) => {
    const order = {
      TYPE_CHANGED: 0,
      GEOMETRY_CHANGED: 1,
      MOVED: 2,
      ONLY_A: 3,
      ONLY_B: 4,
      MATCHED: 5,
    };

    const a = order[r1.status] ?? 9;
    const b = order[r2.status] ?? 9;

    if (a !== b) return a - b;

    const p1 = r1.a?.pipeline || r1.b?.pipeline || '';
    const p2 = r2.a?.pipeline || r2.b?.pipeline || '';

    return String(p1).localeCompare(String(p2));
  });

  return {
    schemaVersion: 'xml-compare-report/v2',
    generatedAt: new Date().toISOString(),
    files: {
      A: datasetA?.sourceFileName || '',
      B: datasetB?.sourceFileName || '',
    },
    tolerance,
    counts,
    rows,
  };
}
```

# X9 — configurable matching strategy

X8 added tolerance, but matching still needs user control. In real XML, some files have stable `RefNo/SeqNo`, some have only geometry, and some have shifted origins. X9 adds:

```text
Match By:
1. Ref + Seq
2. Ref only
3. Pipeline + Type + Geometry
4. Geometry only
```

This prevents false `ONLY_A / ONLY_B` results when XML identity fields differ.

```js
export const XML_COMPARE_MATCH_STRATEGIES = Object.freeze({
  REF_SEQ: 'REF_SEQ',
  REF_ONLY: 'REF_ONLY',
  PIPELINE_TYPE_GEOMETRY: 'PIPELINE_TYPE_GEOMETRY',

export const XML_COMPARE_DEFAULT_MATCH_STRATEGY = XML_COMPARE_MATCH_STRATEGIES.REF_SEQ;

export function normalizeXmlCompareMatchStrategy(value) {
  const v = String(value || '').trim().toUpperCase();

  if (v === XML_COMPARE_MATCH_STRATEGIES.REF_ONLY) return XML_COMPARE_MATCH_STRATEGIES.REF_ONLY;
  if (v === XML_COMPARE_MATCH_STRATEGIES.PIPELINE_TYPE_GEOMETRY) return XML_COMPARE_MATCH_STRATEGIES.PIPELINE_TYPE_GEOMETRY;
  if (v === XML_COMPARE_MATCH_STRATEGIES.GEOMETRY_ONLY) return XML_COMPARE_MATCH_STRATEGIES.GEOMETRY_ONLY;

  return XML_COMPARE_MATCH_STRATEGIES.REF_SEQ;
}

function componentCompareSeq(component) {
  const attrs = component?.attributes || {};
  const source = component?.source || {};

  return clean(
    component?.seqNo ||
    source.seqNo ||
    attrs['COMPONENT-ATTRIBUTE98'] ||
    attrs.COMPONENT_ATTRIBUTE98 ||
    attrs.SEQ_NO ||
    ''
  );
}

function componentCompareGeometryBucket(component, bucketMm = 25) {
  const p1 = pointOf(component, 0) || {};
  const p2 = component?.points?.[1] || {};

  const bucket = Math.max(1, Number(bucketMm) || 25);

  return [
    Math.round(Number(p1.x || 0) / bucket),
    Math.round(Number(p1.y || 0) / bucket),
    Math.round(Number(p1.z || 0) / bucket),
    Math.round(Number(p2.x || 0) / bucket),
    Math.round(Number(p2.y || 0) / bucket),
```

```js
  const pipeline = componentComparePipeline(component);
```

```md
# XML Compare X2-X10 Notes

## Scope

This module is for the normal 3D Viewer / XML geometry compare workflow.

It is not part of:

- 3D RVM Viewer
- RVM PCF readiness
- RVM tag XML import/export
- RVM support symbol scaling

## Features

- Import XML A and XML B.
- Render both in the same canvas.
- Different color palettes for A and B.
- Numeric offset/origin tools.
- Pick-origin tools.
- Tree hierarchy: dataset -> pipeline -> type -> component.
- Tree/canvas selection synchronization.
- Multi-select in tree.
- Visibility control.
- Compare modes:
  - Overlay
  - A only
  - B only
  - Ghost overlay
  - Changed color
- Tolerance-based compare.
- Matching strategies:
  - Ref + Seq
  - Ref only
  - Pipeline + Type + Geometry
  - Geometry only
- JSON and CSV report export.

## Known Limitations

1. No semantic piping-change classification yet.
   Example: a reducer changed to valve is identified as TYPE_CHANGED, but not described as engineering intent.

2. Geometry matching is component-level, not full-topology-level.
   It does not yet compare branch connectivity, tee/olet parent-child relation, or support-to-pipe attachment logic.

3. Origin alignment is point-based only.
   Three-point alignment / rotation alignment is not included yet.

4. Marquee selection is not included yet.
   Tree multiselect and canvas click selection are included.

5. Difference report is deterministic except generatedAt timestamp.

## Recommended Next Phase

X11:

- Three-point alignment
- Rotation alignment
- Topology-aware compare
- Support attachment compare
- Marquee canvas selection
```

```js
 *
 * Source-coordinate rigid alignment helpers.

function n(value, fallback = 0) {
  const x = Number(value);
  return Number.isFinite(x) ? x : fallback;
}

export function normalizeXmlComparePoint(p = {}) {
  return {
    x: n(p.x),
    y: n(p.y),
    z: n(p.z),
  };
}

function sub(a, b) {
  return {
    x: a.x - b.x,
    y: a.y - b.y,
    z: a.z - b.z,
  };
}

function add(a, b) {
  return {
    x: a.x + b.x,
    y: a.y + b.y,
    z: a.z + b.z,
  };
}

function scale(v, s) {
  return {
    x: v.x * s,
    y: v.y * s,
    z: v.z * s,
  };
}

function dot(a, b) {
  return a.x * b.x + a.y * b.y + a.z * b.z;
}

function cross(a, b) {
  return {
    x: a.y * b.z - a.z * b.y,
    y: a.z * b.x - a.x * b.z,
    z: a.x * b.y - a.y * b.x,
  };
}

function len(v) {
  return Math.sqrt(dot(v, v));
}

function norm(v, fallback = { x: 1, y: 0, z: 0 }) {
  const l = len(v);
  if (l < 1e-9) return { ...fallback };
  return scale(v, 1 / l);
}

function matIdentity() {
  return [
    1, 0, 0,
    0, 1, 0,
    0, 0, 1,
  ];
}

function matTranspose(m) {
  return [
    m[0], m[3], m[6],
    m[1], m[4], m[7],
    m[2], m[5], m[8],
  ];
}

function matMul(a, b) {
  return [
    a[0] * b[0] + a[1] * b[3] + a[2] * b[6],
    a[0] * b[1] + a[1] * b[4] + a[2] * b[7],
    a[0] * b[2] + a[1] * b[5] + a[2] * b[8],

    a[3] * b[0] + a[4] * b[3] + a[5] * b[6],
    a[3] * b[1] + a[4] * b[4] + a[5] * b[7],
    a[3] * b[2] + a[4] * b[5] + a[5] * b[8],

    a[6] * b[0] + a[7] * b[3] + a[8] * b[6],
    a[6] * b[1] + a[7] * b[4] + a[8] * b[7],
    a[6] * b[2] + a[7] * b[5] + a[8] * b[8],
  ];
}

function matVec(m, p) {
  return {
    x: m[0] * p.x + m[1] * p.y + m[2] * p.z,
    y: m[3] * p.x + m[4] * p.y + m[5] * p.z,
    z: m[6] * p.x + m[7] * p.y + m[8] * p.z,
  };
}

function basisFrom3Points(points = []) {
  const p0 = normalizeXmlComparePoint(points[0]);
  const p1 = normalizeXmlComparePoint(points[1]);
  const p2 = normalizeXmlComparePoint(points[2]);

  const xAxis = norm(sub(p1, p0), { x: 1, y: 0, z: 0 });
  const plane = sub(p2, p0);
  const zAxis = norm(cross(xAxis, plane), { x: 0, y: 0, z: 1 });
  const yAxis = norm(cross(zAxis, xAxis), { x: 0, y: 1, z: 0 });

  // Row-major matrix with basis vectors as columns.
  return {
    origin: p0,
    matrix: [
      xAxis.x, yAxis.x, zAxis.x,
      xAxis.y, yAxis.y, zAxis.y,
      xAxis.z, yAxis.z, zAxis.z,
    ],
  };
}

function rotationFromOneVector(sourceVector, targetVector) {
  const a = norm(sourceVector);
  const b = norm(targetVector);

  const v = cross(a, b);
  const c = Math.max(-1, Math.min(1, dot(a, b)));
  const s = len(v);

  if (s < 1e-9) {
    return c > 0 ? matIdentity() : [
      -1, 0, 0,
       0, 1, 0,
       0, 0, -1,
    ];
  }

  const vx = [
     0, -v.z, v.y,
     v.z, 0, -v.x,
    -v.y, v.x, 0,
  ];

  const vx2 = matMul(vx, vx);
  const k = (1 - c) / (s * s);

  return matIdentity().map((value, index) =>
    value + vx[index] + vx2[index] * k
  );
}

export function createXmlCompareTranslationAlignment(sourcePoint, targetPoint) {
  const s = normalizeXmlComparePoint(sourcePoint);
  const t = normalizeXmlComparePoint(targetPoint);

  return {
    kind: 'TRANSLATION',
    rotation: matIdentity(),
    offset: sub(t, s),
  };
}

export function createXmlCompareTwoPointAlignment(sourcePoints = [], targetPoints = []) {
  const s0 = normalizeXmlComparePoint(sourcePoints[0]);
  const s1 = normalizeXmlComparePoint(sourcePoints[1]);
  const t0 = normalizeXmlComparePoint(targetPoints[0]);
  const t1 = normalizeXmlComparePoint(targetPoints[1]);

  const rotation = rotationFromOneVector(sub(s1, s0), sub(t1, t0));
  const rotatedSourceOrigin = matVec(rotation, s0);

  return {
    kind: 'TWO_POINT',
    rotation,
    offset: sub(t0, rotatedSourceOrigin),
  };
}

export function createXmlCompareThreePointAlignment(sourcePoints = [], targetPoints = []) {
  const sourceBasis = basisFrom3Points(sourcePoints);
  const targetBasis = basisFrom3Points(targetPoints);

  const rotation = matMul(targetBasis.matrix, matTranspose(sourceBasis.matrix));
  const rotatedSourceOrigin = matVec(rotation, sourceBasis.origin);

  return {
    kind: 'THREE_POINT',
    rotation,
    offset: sub(targetBasis.origin, rotatedSourceOrigin),
  };
}

export function createXmlCompareAlignmentTransform(sourcePoints = [], targetPoints = []) {
  const usable = Math.min(sourcePoints.length, targetPoints.length);

  if (usable >= 3) {
    return createXmlCompareThreePointAlignment(sourcePoints.slice(0, 3), targetPoints.slice(0, 3));
  }

  if (usable >= 2) {
    return createXmlCompareTwoPointAlignment(sourcePoints.slice(0, 2), targetPoints.slice(0, 2));
  }

  if (usable >= 1) {
    return createXmlCompareTranslationAlignment(sourcePoints[0], targetPoints[0]);
  }

  return {
    kind: 'IDENTITY',
    rotation: matIdentity(),
    offset: { x: 0, y: 0, z: 0 },
  };
}

export function transformXmlComparePoint(point, alignmentTransform = null) {
  const p = normalizeXmlComparePoint(point);

  if (!alignmentTransform?.rotation) return p;

  const r = matVec(alignmentTransform.rotation, p);
  const o = normalizeXmlComparePoint(alignmentTransform.offset || {});

  return add(r, o);
}

export function isValidXmlCompareAlignment(transform) {
  return (
    transform &&
    Array.isArray(transform.rotation) &&
    transform.rotation.length === 9 &&
    transform.offset &&
    ['IDENTITY', 'TRANSLATION', 'TWO_POINT', 'THREE_POINT'].includes(transform.kind)
  );
}
```

```js
function topologyNodeKey(point, toleranceMm = 5) {
  if (!point) return 'NO_POINT';

  const bucket = Math.max(1, Number(toleranceMm) || 5);

  return [
    Math.round(Number(point.x || 0) / bucket),
    Math.round(Number(point.y || 0) / bucket),
    Math.round(Number(point.z || 0) / bucket),
  ].join(',');
}

function topologyComponentEdges(dataset, tolerance = {}) {
  const tol = normalizeXmlCompareTolerance(tolerance);
  const edges = new Set();
  const nodes = new Set();

  for (const component of dataset?.components || []) {
    const p1 = transformedPointOf(component, 0, dataset?.transform);
    const p2 = transformedPointOf(component, 1, dataset?.transform);

    if (!p1 || !p2) continue;

    const a = topologyNodeKey(p1, tol.positionMm);
    const b = topologyNodeKey(p2, tol.positionMm);

    nodes.add(a);
    nodes.add(b);

    edges.add([a, b].sort().join('::'));
  }

  return {
    nodeCount: nodes.size,
    edgeCount: edges.size,
    edges,
  };
}

export function compareXmlCompareTopology(datasetA, datasetB, tolerance = {}) {
  const a = topologyComponentEdges(datasetA, tolerance);
  const b = topologyComponentEdges(datasetB, tolerance);

  let matchedEdges = 0;
  let onlyAEdges = 0;
  let onlyBEdges = 0;

  for (const edge of a.edges) {
    if (b.edges.has(edge)) matchedEdges += 1;
    else onlyAEdges += 1;
  }

  for (const edge of b.edges) {
    if (!a.edges.has(edge)) onlyBEdges += 1;
  }

  return {
    tolerance: normalizeXmlCompareTolerance(tolerance),
    A: {
      nodeCount: a.nodeCount,
      edgeCount: a.edgeCount,
    },
    B: {
      nodeCount: b.nodeCount,
      edgeCount: b.edgeCount,
    },
    matchedEdges,
    onlyAEdges,
    onlyBEdges,
    changedEdges: onlyAEdges + onlyBEdges,
  };
}
```

```js
function componentEndpointPoints(component) {
  const points = Array.isArray(component?.points) ? component.points : [];

  return {
    p1: points[0] || component?.coOrds || component?.coord || null,
    p2: points[1] || null,
    branch: component?.branchPoint || component?.bp || component?.source?.branchPoint || null,
    centre: component?.centrePoint || component?.centerPoint || component?.cp || component?.source?.centrePoint || null,
  };
}

function componentAttachPoint(component) {
  const attrs = component?.attributes || {};

  return (
    component?.supportCoord ||
    component?.coOrds ||
    component?.coord ||
    component?.points?.[0] ||
    component?.source?.supportCoord ||
    attrs.SUPPORT_COORD ||
    null
  );
}

function nearestComponentDistance(point, components = []) {
  let best = {
    component: null,
    distanceMm: Number.POSITIVE_INFINITY,
  };

  if (!point) return best;

  for (const component of components || []) {
    const pts = componentEndpointPoints(component);
    const candidates = [pts.p1, pts.p2, pts.branch, pts.centre].filter(Boolean);

    for (const candidate of candidates) {
      const d = dist3(point, candidate);
      if (d < best.distanceMm) {
        best = {
          component,
          distanceMm: d,
        };
      }
    }
  }

  return best;
}

function isSupportComponent(component) {
  const type = componentCompareType(component);
  const text = [
    type,
    component?.fixingAction,
    component?.attributes?.SUPPORT_KIND,
    component?.attributes?.SUPPORT_TYPE,
    component?.attributes?.['SUPPORT-TYPE'],
    component?.attributes?.SKEY,
  ].map(v => String(v || '').toUpperCase()).join(' ');

  return (
    type === 'SUPPORT' ||
    type === 'ANCI' ||
    /\bSUPPORT\b|\bHANGER\b|\bHANG\b|\bGUIDE\b|\bGUI\b|\bLIM\b|\bLIMIT\b|\bREST\b|\bANCHOR\b/.test(text)
  );
}

function isBranchComponent(component) {
  const type = componentCompareType(component);
  const text = [
    type,
    component?.attributes?.SKEY,
    component?.attributes?.['COMPONENT-TYPE'],
    component?.attributes?.COMPONENT_TYPE,
  ].map(v => String(v || '').toUpperCase()).join(' ');

  return (
    type === 'TEE' ||
    type === 'OLET' ||
    type.includes('TEE') ||
    type.includes('OLET') ||
    /\bTEE\b|\bOLET\b|\bWELDOLET\b|\bSOCKOLET\b|\bTHREADOLET\b|\bBRANCH\b/.test(text)
  );
}

function nonSupportPipeLikeComponents(dataset) {
  return (dataset?.components || []).filter(component => {
    const type = componentCompareType(component);
    return (
      !isSupportComponent(component) &&
      ['PIPE', 'BEND', 'ELBOW', 'TEE', 'OLET', 'REDUCER', 'VALVE', 'FLANGE'].some(t => type.includes(t))
    );
  });
}

function componentInfoForIssue(component, datasetId) {
  if (!component) return null;

  return {
    datasetId,
    componentId: component.id || '',
    type: componentCompareType(component),
    pipeline: componentComparePipeline(component),
    label: componentCompareLabel(component),
  };
}

export function buildXmlCompareSupportAttachmentIssues(dataset, datasetId, tolerance = {}) {
  const tol = normalizeXmlCompareTolerance(tolerance);
  const supports = (dataset?.components || []).filter(isSupportComponent);
  const pipeLike = nonSupportPipeLikeComponents(dataset);

  const issues = [];

  for (const support of supports) {
    const attach = componentAttachPoint(support);
    const nearest = nearestComponentDistance(attach, pipeLike);

    if (!nearest.component || nearest.distanceMm > tol.positionMm) {
      issues.push({
        code: 'SUPPORT-ATTACHMENT-MISSING',
        severity: 'WARNING',
        datasetId,
        componentId: support.id || '',
        component: componentInfoForIssue(support, datasetId),
        nearestComponent: nearest.component ? componentInfoForIssue(nearest.component, datasetId) : null,
        distanceMm: Number.isFinite(nearest.distanceMm) ? nearest.distanceMm : null,
        toleranceMm: tol.positionMm,
        message: `Support is not attached to nearby pipe geometry within ${tol.positionMm} mm.`,
      });
    }
  }

  return issues;
}

export function buildXmlCompareBranchConnectivityIssues(dataset, datasetId, tolerance = {}) {
  const tol = normalizeXmlCompareTolerance(tolerance);
  const branches = (dataset?.components || []).filter(isBranchComponent);
  const pipeLike = nonSupportPipeLikeComponents(dataset);

  const issues = [];

  for (const branch of branches) {
    const pts = componentEndpointPoints(branch);
    const branchPoint = pts.branch || pts.centre || pts.p1;

    const nearest = nearestComponentDistance(
      branchPoint,
      pipeLike.filter(component => component.id !== branch.id)
    );

    if (!nearest.component || nearest.distanceMm > tol.positionMm) {
      issues.push({
        code: componentCompareType(branch).includes('OLET')
          ? 'OLET-BRANCH-CONNECTIVITY-CHANGED'
          : 'TEE-BRANCH-CONNECTIVITY-CHANGED',
        severity: 'ERROR',
        datasetId,
        componentId: branch.id || '',
        component: componentInfoForIssue(branch, datasetId),
        nearestComponent: nearest.component ? componentInfoForIssue(nearest.component, datasetId) : null,
        distanceMm: Number.isFinite(nearest.distanceMm) ? nearest.distanceMm : null,
        toleranceMm: tol.positionMm,
        message: `${componentCompareType(branch)} branch point is not connected to nearby pipe geometry within ${tol.positionMm} mm.`,
      });
    }
  }

  return issues;
}

export function buildXmlCompareTopologyIssueReport(datasetA, datasetB, tolerance = {}) {
  const tol = normalizeXmlCompareTolerance(tolerance);

  const issues = [
    ...buildXmlCompareSupportAttachmentIssues(datasetA, 'A', tol),
    ...buildXmlCompareSupportAttachmentIssues(datasetB, 'B', tol),
    ...buildXmlCompareBranchConnectivityIssues(datasetA, 'A', tol),
    ...buildXmlCompareBranchConnectivityIssues(datasetB, 'B', tol),
  ];

  const counts = issues.reduce((acc, issue) => {
    acc.total += 1;
    acc[issue.code] = (acc[issue.code] || 0) + 1;
    acc.bySeverity[issue.severity] = (acc.bySeverity[issue.severity] || 0) + 1;
    return acc;
  }, {
    total: 0,
    bySeverity: {},
  });

  return {
    schemaVersion: 'xml-compare-topology-issues/v1',
    tolerance: tol,
    counts,
    issues,
  };
}
```

```js
function topologyIssuesAll() {
  return _diffReport?.topologyIssues?.issues || [];
}

function topologyIssueText(issue) {
  return [
    issue?.code,
    issue?.severity,
    issue?.datasetId,
    issue?.componentId,
    issue?.component?.type,
    issue?.component?.pipeline,
    issue?.component?.label,
    issue?.nearestComponent?.componentId,
    issue?.nearestComponent?.type,
    issue?.message,
  ].map(v => String(v || '').toLowerCase()).join(' ');
}

function topologyIssuePassesFilters(issue) {
  if (!issue) return false;

  if (_topologyIssueFilters.severity !== 'ALL' && issue.severity !== _topologyIssueFilters.severity) {
    return false;
  }

  if (_topologyIssueFilters.datasetId !== 'ALL' && issue.datasetId !== _topologyIssueFilters.datasetId) {
    return false;
  }

  if (_topologyIssueFilters.code !== 'ALL' && issue.code !== _topologyIssueFilters.code) {
    return false;
  }

  const search = String(_topologyIssueFilters.search || '').trim().toLowerCase();
  if (search && !topologyIssueText(issue).includes(search)) {
    return false;
  }

  return true;
}

function filteredTopologyIssues() {
  return topologyIssuesAll().filter(topologyIssuePassesFilters);
}

function topologyIssueComponentIds(issues = filteredTopologyIssues()) {
  return [
    ...new Set(
      issues
        .map(issue => issue?.componentId)
        .filter(Boolean)
    ),
  ];
}

function updateTopologyCodeFilter(container) {
  const select = container.querySelector('[data-xml-compare-topology-filter="code"]');
  if (!select) return;

  const previous = _topologyIssueFilters.code || 'ALL';
  const codes = [
    ...new Set(topologyIssuesAll().map(issue => issue.code).filter(Boolean)),
  ].sort();

  select.innerHTML = `
    <option value="ALL">All codes</option>
    ${codes.map(code => `<option value="${esc(code)}">${esc(code)}</option>`).join('')}
  `;

  select.value = codes.includes(previous) ? previous : 'ALL';
  _topologyIssueFilters.code = select.value;
}

function updateTopologyFilterControls(container) {
  for (const [key, value] of Object.entries(_topologyIssueFilters)) {
    const control = container.querySelector(`[data-xml-compare-topology-filter="${CSS.escape(key)}"]`);
    if (control) control.value = value;
  }
}

function updateTopologyFilterSummary(container) {
  const el = container.querySelector('[data-xml-compare-topology-filter-summary]');
  if (!el) return;

  const shown = filteredTopologyIssues().length;
  const total = topologyIssuesAll().length;
  const ids = topologyIssueComponentIds();

  el.textContent = `${shown} shown / ${total} total · ${ids.length} component(s)`;
}
```

```js
function renderTopologyIssues(container) {
```

```js
function hideFilteredTopologyIssues(container) {
  const ids = topologyIssueComponentIds();

  if (!ids.length) {
    notify({ type: 'warning', message: 'No filtered topology issue components to hide.' });
    return;
  }

  _hiddenComponentIds = setVisibilityForIds(_hiddenComponentIds, ids, false);
  applyVisibilityToRenderer(container);
  persistXmlCompareUiState();

  notify({
    type: 'info',
    message: `Hidden ${ids.length} topology issue component(s).`,
  });
}

function isolateFilteredTopologyIssues(container) {
  const ids = topologyIssueComponentIds();

  if (!ids.length) {
    notify({ type: 'warning', message: 'No filtered topology issue components to isolate.' });
    return;
  }

  const keep = new Set(ids);
  const allIds = [];

  for (const dataset of loadedDatasets()) {
    for (const component of dataset.components || []) {
      allIds.push(component.id);
    }
  }

  _hiddenComponentIds = new Set(allIds.filter(id => !keep.has(id)));
  applyVisibilityToRenderer(container);
  persistXmlCompareUiState();

  notify({
    type: 'info',
    message: `Isolated ${ids.length} topology issue component(s).`,
  });
}

function exportFilteredTopologyIssuesCsv(container) {
  const issues = filteredTopologyIssues();

  const header = [
    'Code',
    'Severity',
    'Dataset',
    'Component Id',
    'Type',
    'Pipeline',
    'Label',
    'Nearest Component Id',
    'Distance mm',
    'Tolerance mm',
    'Message',
  ];

  const lines = [header.map(csvCellForUi).join(',')];

  for (const issue of issues) {
    lines.push([
      issue.code,
      issue.severity,
      issue.datasetId,
      issue.componentId,
      issue.component?.type || '',
      issue.component?.pipeline || '',
      issue.component?.label || '',
      issue.nearestComponent?.componentId || '',
      issue.distanceMm ?? '',
      issue.toleranceMm ?? '',
      issue.message || '',
    ].map(csvCellForUi).join(','));
  }

  downloadTextFile(
    'xml-compare-filtered-topology-issues.csv',
    `${lines.join('\n')}\n`,
    'text/csv'
  );

  notify({
    type: 'info',
    message: `Exported ${issues.length} filtered topology issue(s).`,
  });
}

function csvCellForUi(value) {
  const text = String(value ?? '');
  if (/[",\n\r]/.test(text)) {
    return `"${text.replaceAll('"', '""')}"`;
  }
  return text;
}
```

```js
import assert from 'assert/strict';

function passes(issue, filters) {
  if (filters.severity !== 'ALL' && issue.severity !== filters.severity) return false;
  if (filters.datasetId !== 'ALL' && issue.datasetId !== filters.datasetId) return false;
  if (filters.code !== 'ALL' && issue.code !== filters.code) return false;

  const search = String(filters.search || '').toLowerCase();
  if (search) {
    const text = [
      issue.code,
      issue.severity,
      issue.datasetId,
      issue.componentId,
      issue.component?.type,
      issue.component?.pipeline,
      issue.component?.label,
      issue.message,
    ].map(v => String(v || '').toLowerCase()).join(' ');

    if (!text.includes(search)) return false;
  }

  return true;
}

function run() {
  console.log('--- xml-compare-x13-topology-filter-behavior.test.js ---');

  const issues = [
    {
      code: 'SUPPORT-ATTACHMENT-MISSING',
      severity: 'WARNING',
      datasetId: 'A',
      componentId: 'S1',
      component: { type: 'SUPPORT', pipeline: 'LINE-A', label: 'SUP-1' },
      message: 'Support missing',
    },
    {
      code: 'TEE-BRANCH-CONNECTIVITY-CHANGED',
      severity: 'ERROR',
      datasetId: 'B',
      componentId: 'T1',
      component: { type: 'TEE', pipeline: 'LINE-B', label: 'TEE-1' },
      message: 'Tee branch changed',
    },
  ];

  assert.equal(
    issues.filter(issue => passes(issue, {
      severity: 'ERROR',
      datasetId: 'ALL',
      code: 'ALL',
      search: '',
    })).length,
    1
  );

  assert.equal(
    issues.filter(issue => passes(issue, {
      severity: 'ALL',
      datasetId: 'A',
      code: 'ALL',
      search: '',
    })).length,
    1
  );

  assert.equal(
    issues.filter(issue => passes(issue, {
      severity: 'ALL',
      datasetId: 'ALL',
      code: 'SUPPORT-ATTACHMENT-MISSING',
      search: '',
    })).length,
    1
  );

  assert.equal(
    issues.filter(issue => passes(issue, {
      severity: 'ALL',
      datasetId: 'ALL',
      code: 'ALL',
      search: 'line-b',
    })).length,
    1
  );

  console.log('[PASS] XML Compare X13 topology filter behavior passed.');
}

try {
  run();
} catch (error) {
  console.error('[FAIL] XML Compare X13 topology filter behavior failed.');
```
