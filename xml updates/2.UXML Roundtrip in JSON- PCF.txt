function simulateGeneratePcfButtonContract({
  topologyMode,
  rows,
  readinessGate,
  allowPartialExport = false,
}) {
  const guard = evaluateRvmPcfExportGuard({
    topologyMode,
    rows,
    readinessGate,
    allowPartialExport,
  });

  if (!guard.allowed) {
    return {
      clicked: true,
      allowed: false,
      blocked: true,
      guard,
      message: formatRvmPcfExportGuardMessage(guard),
      legacyPcfRouteInvoked: false,
      uxmlGeneratedPcf: false,
      mastersResolvedByUxml: false,
    };
  }

  const assertedGuard = assertRvmPcfExportAllowed({
    topologyMode,
    rows,
    readinessGate,
    allowPartialExport,
  });

  return {
    clicked: true,
    allowed: true,
    blocked: false,
    guard: assertedGuard,
    message: formatRvmPcfExportGuardMessage(assertedGuard),
    legacyPcfRouteInvoked: true,
    uxmlGeneratedPcf: false,
    mastersResolvedByUxml: false,
  };
}

    const rows = rowFixture();

      rows,

    });

    expect(result.allowed).toBe(true);
    expect(result.blocked).toBe(false);
    expect(result.legacyPcfRouteInvoked).toBe(true);

    expect(result.guard.allowed).toBe(true);
    expect(result.guard.outputBridgeReady).toBe(true);
    expect(result.guard.acceptedConnectionCount).toBeGreaterThan(0);

    expect(result.guard.legacyRoutingContinues).toBe(true);
    expect(result.guard.mastersDeferredToLegacyRoute).toBe(true);
    expect(result.guard.pcfEmitterDeferredToLegacyRoute).toBe(true);

    expect(result.uxmlGeneratedPcf).toBe(false);
    expect(result.mastersResolvedByUxml).toBe(false);

      allowPartialExport: true,
    });

      legacyPcfRouteInvoked: result.legacyPcfRouteInvoked,
      mastersOwnedByLegacyRoute: result.guard.mastersDeferredToLegacyRoute === true,
      pcfEmitterOwnedByLegacyRoute: result.guard.pcfEmitterDeferredToLegacyRoute === true,
      uxmlGeneratedPcf: result.uxmlGeneratedPcf,
      mastersResolvedByUxml: result.mastersResolvedByUxml,
    };

    expect(ownership).toEqual({
      topologyBuiltByUxml: true,
      evidenceReturnedToLegacyRows: true,
      generateButtonAllowedByGuard: true,
      legacyPcfRouteInvoked: true,
      mastersOwnedByLegacyRoute: true,
      pcfEmitterOwnedByLegacyRoute: true,
      uxmlGeneratedPcf: false,
      mastersResolvedByUxml: false,

function geometryFingerprint(rows) {
  return rows.map(row => ({
    id: row.id,
    type: row.type,
    pipelineRef: row.pipelineRef,
    lineNo: row.lineNo,
    refNo: row.refNo,
    seqNo: row.seqNo,
    ep1: row.ep1 ?? null,
    ep2: row.ep2 ?? null,
    cp: row.cp ?? null,
    bp: row.bp ?? null,
    supportCoord: row.supportCoord ?? null,
  }));
}

/**
 * Local test stub for the existing legacy PCF writer ownership boundary.
 *
 * This is intentionally not a production writer.
 * The test only proves that PCF text is created after export guard passes,
 * from legacyRows, not from the UXML topology bridge.
 */
function simulateLegacyPcfWriter({
  rows,
  guard,
}) {
  if (!guard?.allowed) {
    throw new Error('Legacy PCF writer must not run before export guard passes.');
  }

  const byPipelineRef = {};

  for (const row of rows || []) {
    const pipelineRef = row.pipelineRef || 'UNKNOWN-PIPELINE';

    if (!byPipelineRef[pipelineRef]) {
      byPipelineRef[pipelineRef] = [
        'ISOGEN-FILES ISOGEN.FLS',
        `PIPELINE-REFERENCE ${pipelineRef}`,
        `    ATTRIBUTE1 ${row.lineNo || ''}`,
      ];
    }

    byPipelineRef[pipelineRef].push(
      `${row.type || 'COMPONENT'} ${row.id || row.name || row.rowNo || ''}`
    );

    if (row.ep1) byPipelineRef[pipelineRef].push(`    END-POINT ${row.ep1}`);
    if (row.ep2) byPipelineRef[pipelineRef].push(`    END-POINT ${row.ep2}`);
    if (row.cp) byPipelineRef[pipelineRef].push(`    CENTRE-POINT ${row.cp}`);
    if (row.bp) byPipelineRef[pipelineRef].push(`    BRANCH-POINT ${row.bp}`);

    if (row._uxmlAcceptedTopologyCount > 0) {
      byPipelineRef[pipelineRef].push(
        `    ATTRIBUTE99 UXML-TOPOLOGY-EVIDENCE:${row._uxmlAcceptedTopologyCount}`
      );
    }
  }

  const pcfTextByPipelineRef = Object.fromEntries(
    Object.entries(byPipelineRef).map(([pipelineRef, lines]) => [
      pipelineRef,
      `${lines.join('\n')}\n`,
    ])
  );

  return {
    schema: 'rjp-3-legacy-pcf-writer-stub/v1',
    owner: 'LEGACY_JSON_RVM_TO_PCF_ROUTE',
    generatedByUxml: false,
    mastersResolvedByUxml: false,
    pcfTextByPipelineRef,
    pipelineCount: Object.keys(pcfTextByPipelineRef).length,
  };
}

describe('RJP-3 PCF text output ownership smoke', () => {
  it('keeps UXML topology bridge topology-only and lets legacy route own PCF text', () => {
    const rows = rowFixture();
    const beforeRows = JSON.stringify(rows);
    const beforeGeometry = geometryFingerprint(rows);

    const uxmlResult = runUxmlTopologyForRvmRows(rows, {
      connectToleranceMm: 6,
      maxRayLengthMm: 500,
      tubeToleranceMm: 12,
      allowPartialExport: true,
      name: 'rjp-3-pcf-text-output-ownership',
    });

    expect(JSON.stringify(rows)).toBe(beforeRows);
    expect(geometryFingerprint(rows)).toEqual(beforeGeometry);

    expect(uxmlResult.schema).toBe('rvm-pcf-uxml-topology-bridge/v1');
    expect(uxmlResult.topologyMode).toBe(RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY);

    expect(uxmlResult.topologyDecision).toBeTruthy();
    expect(uxmlResult.acceptedTopologyHandoff).toBeTruthy();
    expect(uxmlResult.readinessGate).toBeTruthy();
    expect(uxmlResult.legacyRows).toHaveLength(rows.length);

    expect(uxmlResult.generatedPcf).toBeUndefined();
    expect(uxmlResult.pcfTextByPipelineRef).toBeUndefined();
    expect(uxmlResult.masterResolution).toBeUndefined();
    expect(uxmlResult.masterResolutionRequests).toBeUndefined();

    expect(uxmlResult.readinessGate.summary.legacyRoutingContinues).toBe(true);
    expect(uxmlResult.readinessGate.summary.mastersDeferredToLegacyRoute).toBe(true);
    expect(uxmlResult.readinessGate.summary.pcfEmitterDeferredToLegacyRoute).toBe(true);

    const guard = assertRvmPcfExportAllowed({
      topologyMode: RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY,
      rows: uxmlResult.legacyRows,
      readinessGate: uxmlResult.readinessGate,
      allowPartialExport: true,
    });

    expect(guard.allowed).toBe(true);
    expect(guard.outputBridgeReady).toBe(true);
    expect(guard.legacyRoutingContinues).toBe(true);
    expect(guard.mastersDeferredToLegacyRoute).toBe(true);
    expect(guard.pcfEmitterDeferredToLegacyRoute).toBe(true);

    const legacyWriterResult = simulateLegacyPcfWriter({
      rows: uxmlResult.legacyRows,
      guard,
    });

    expect(legacyWriterResult.owner).toBe('LEGACY_JSON_RVM_TO_PCF_ROUTE');
    expect(legacyWriterResult.generatedByUxml).toBe(false);
    expect(legacyWriterResult.mastersResolvedByUxml).toBe(false);
    expect(legacyWriterResult.pipelineCount).toBe(1);

    expect(Object.keys(legacyWriterResult.pcfTextByPipelineRef)).toEqual(['/RJP-3001']);

    const pcfText = legacyWriterResult.pcfTextByPipelineRef['/RJP-3001'];

    expect(pcfText).toContain('ISOGEN-FILES');
    expect(pcfText).toContain('PIPELINE-REFERENCE /RJP-3001');
    expect(pcfText).toContain('PIPE PIPE-1');
    expect(pcfText).toContain('VALVE VALVE-1');
    expect(pcfText).toContain('PIPE PIPE-2');
    expect(pcfText).toContain('FLANGE FLANGE-1');

    expect(pcfText).toContain('UXML-TOPOLOGY-EVIDENCE');

    expect(uxmlResult.generatedPcf).toBeUndefined();
    expect(uxmlResult.pcfTextByPipelineRef).toBeUndefined();
  });

  it('does not allow legacy PCF text ownership before UXML export guard passes', () => {
    const rows = rowFixture();

    const guard = evaluateRvmPcfExportGuard({
      topologyMode: RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY,
      rows,
      readinessGate: null,
      allowPartialExport: false,
    });

    expect(guard.allowed).toBe(false);

    expect(() => simulateLegacyPcfWriter({
      rows,
      guard,
    })).toThrow('Legacy PCF writer must not run before export guard passes.');
  });

  it('documents PCF ownership boundary as a deterministic contract', () => {
    const rows = rowFixture();

    const uxmlResult = runUxmlTopologyForRvmRows(rows, {
      allowPartialExport: true,
    });

    const guard = assertRvmPcfExportAllowed({
      topologyMode: RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY,
      rows: uxmlResult.legacyRows,
      readinessGate: uxmlResult.readinessGate,
      allowPartialExport: true,
    });

    const legacyWriterResult = simulateLegacyPcfWriter({
      rows: uxmlResult.legacyRows,
      guard,
    });

    const contract = {
      topologyMode: RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY,
      topologyGeneratedByUxml: !!uxmlResult.topologyDecision,
      topologyEvidenceReturnedToLegacyRows: uxmlResult.legacyRows.some(row =>
        row._topologyMode === RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY ||
        row._uxmlAcceptedTopologyCount > 0
      ),
      exportGuardPassed: guard.allowed === true,
      pcfTextProduced: Object.keys(legacyWriterResult.pcfTextByPipelineRef).length > 0,
      pcfTextOwner: legacyWriterResult.owner,
      pcfTextProducedByUxml: legacyWriterResult.generatedByUxml,
      mastersResolvedByUxml: legacyWriterResult.mastersResolvedByUxml,
      uxmlResultContainsPcfText: uxmlResult.pcfTextByPipelineRef != null,
      uxmlResultContainsMasterResolution: uxmlResult.masterResolution != null,
      sourceCoordinatesMutated: uxmlResult.acceptedTopologyHandoff.summary.coordinatesMutated === true,
    };

    expect(contract).toEqual({
      topologyMode: RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY,
      topologyGeneratedByUxml: true,
      topologyEvidenceReturnedToLegacyRows: true,
      exportGuardPassed: true,
      pcfTextProduced: true,
      pcfTextOwner: 'LEGACY_JSON_RVM_TO_PCF_ROUTE',
      pcfTextProducedByUxml: false,
      mastersResolvedByUxml: false,
      uxmlResultContainsPcfText: false,
      uxmlResultContainsMasterResolution: false,
      sourceCoordinatesMutated: false,
    });
  });
});

function createDom() {

    <!doctype html>

      <body>
        <section id="rvm-pcf-extract-test-host">
          <select data-testid="rvm-pcf-topology-mode-select">

function simulateLegacyPcfText(rows) {
  const byRef = {};

  for (const row of rows) {
    const ref = row.pipelineRef || 'UNKNOWN';
    if (!byRef[ref]) {
      byRef[ref] = [
        'ISOGEN-FILES ISOGEN.FLS',
        `PIPELINE-REFERENCE ${ref}`,
      ];
    }

    byRef[ref].push(`${row.type} ${row.id}`);
    if (row.ep1) byRef[ref].push(`    END-POINT ${row.ep1}`);
    if (row.ep2) byRef[ref].push(`    END-POINT ${row.ep2}`);
  }

  return Object.fromEntries(

function wireTestUi({
  rows,
  onLegacyGenerate = vi.fn(),
}) {
  const state = {
    topologyMode: RVM_PCF_TOPOLOGY_MODES.LEGACY,
    rows,
    readinessGate: null,
    uxmlResult: null,
    pcfTextByPipelineRef: {},
  };

  const select = document.querySelector('[data-testid="rvm-pcf-topology-mode-select"]');
  const runReadiness = document.querySelector('[data-testid="rvm-pcf-run-readiness"]');
  const generate = document.querySelector('[data-testid="rvm-pcf-generate-pcf"]');
  const status = document.querySelector('[data-testid="rvm-pcf-status"]');
  const diagnostics = document.querySelector('[data-testid="rvm-pcf-panel-diagnostics"]');
  const pcfPanel = document.querySelector('[data-testid="rvm-pcf-panel-pcf"]');

  select.addEventListener('change', () => {
    state.topologyMode = select.value === 'uxml-topology'
      ? RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY
      : RVM_PCF_TOPOLOGY_MODES.LEGACY;

    status.textContent = `Topology mode: ${state.topologyMode}`;
  });

  runReadiness.addEventListener('click', () => {
    if (state.topologyMode === RVM_PCF_TOPOLOGY_MODES.UXML_TOPOLOGY) {
      const result = runUxmlTopologyForRvmRows(state.rows, {
        allowPartialExport: true,
        connectToleranceMm: 6,
        maxRayLengthMm: 500,
        tubeToleranceMm: 12,
        name: 'rjp-5-dom-smoke',
      });

      state.uxmlResult = result;
      state.rows = result.legacyRows;
      state.readinessGate = result.readinessGate;

      status.textContent = 'UXML readiness complete';
      diagnostics.textContent = `Accepted topology: ${result.topologyDecision.summary.acceptedConnectionCount}`;
    } else {
      status.textContent = 'Legacy readiness path';
    }
  });

  generate.addEventListener('click', () => {
    const guard = evaluateRvmPcfExportGuard({
      topologyMode: state.topologyMode,
      rows: state.rows,
      readinessGate: state.readinessGate,
      allowPartialExport: true,
    });

    if (!guard.allowed) {
      status.textContent = guard.reason || 'Generate PCF blocked';
      diagnostics.textContent = 'PCF export guard blocked generation';
      return;
    }

    const pcfTextByPipelineRef = simulateLegacyPcfText(state.rows);

    state.pcfTextByPipelineRef = pcfTextByPipelineRef;

    onLegacyGenerate({
      owner: 'LEGACY_JSON_RVM_TO_PCF_ROUTE',
      pcfTextByPipelineRef,
      generatedByUxml: false,
      mastersResolvedByUxml: false,
      guard,
    });

    status.textContent = 'PCF generated by legacy route';
    pcfPanel.textContent = Object.values(pcfTextByPipelineRef).join('\n');
  });

  return {
    state,
    select,
    runReadiness,
    generate,
    status,
    diagnostics,
    pcfPanel,
  };
}

describe('RJP-5 Generate PCF button DOM smoke in UXML topology mode', () => {
  beforeEach(() => {
    createDom();
  });

  it('blocks Generate PCF before UXML readiness is run', () => {
    const onLegacyGenerate = vi.fn();

    const ui = wireTestUi({
      rows: rowFixture(),
      onLegacyGenerate,
    });

    ui.select.value = 'uxml-topology';
    ui.select.dispatchEvent(new Event('change'));

    ui.generate.click();

    expect(onLegacyGenerate).not.toHaveBeenCalled();

    expect(ui.diagnostics.textContent).toContain('guard blocked');
    expect(ui.pcfPanel.textContent).toBe('');
  });

  it('runs readiness then Generate PCF through legacy route', () => {
    const onLegacyGenerate = vi.fn();

    const ui = wireTestUi({
      rows: rowFixture(),
      onLegacyGenerate,
    });

    ui.select.value = 'uxml-topology';
    ui.select.dispatchEvent(new Event('change'));

    ui.runReadiness.click();

    expect(ui.state.uxmlResult).toBeTruthy();

    expect(ui.state.uxmlResult.pcfTextByPipelineRef).toBeUndefined();

    expect(ui.status.textContent).toContain('UXML readiness complete');
    expect(ui.diagnostics.textContent).toContain('Accepted topology');

    ui.generate.click();

    expect(onLegacyGenerate).toHaveBeenCalledTimes(1);

    const payload = onLegacyGenerate.mock.calls[0][0];

    expect(payload.owner).toBe('LEGACY_JSON_RVM_TO_PCF_ROUTE');
    expect(payload.generatedByUxml).toBe(false);
    expect(payload.mastersResolvedByUxml).toBe(false);
    expect(payload.guard.allowed).toBe(true);
    expect(payload.guard.mastersDeferredToLegacyRoute).toBe(true);
    expect(payload.guard.pcfEmitterDeferredToLegacyRoute).toBe(true);

    expect(ui.status.textContent).toContain('legacy route');
    expect(ui.pcfPanel.textContent).toContain('PIPELINE-REFERENCE /RJP-5001');
    expect(ui.pcfPanel.textContent).toContain('PIPE PIPE-1');

  });
